110double SpatialInterpolator::Interpolate(double x, double y) const
111{
112 // Prepare query point
113 double const query_pt[2] = {x, y};
114
115 // Find k nearest neighbors (or all points if fewer than k exist)
116 size_t const k = std::min((size_t) m_k_neighbors, m_cloud.pts.size());
117 std::vector<unsigned int> indices(k);
118 std::vector<double> sq_dists(k); // Squared distances (faster than actual distances)
119
120 unsigned int num_found = m_kdtree->knnSearch(query_pt, k,
121 indices.data(),
122 sq_dists.data());
123
124 if (num_found == 0)
125 throw std::runtime_error("knnSearch found no neighbors");
126
127 // SPECIAL CASE: Query point coincides with an input point
128 // Return exact value to avoid division by zero
129 if (sq_dists[0] < EPSILON)
130 return m_values[indices[0]];
131
132 // Inverse Distance Weighting (IDW)
133 // Formula: result = Σ(w_i * v_i) / Σ(w_i) where w_i = 1/dist_i^power
134 double sum_weights = 0.0;
135 double sum_weighted_values = 0.0;
136
137 if (m_power == 2.0)
138 {
139 // *** OPTIMIZED PATH for power=2.0 ***
140 // Since weight = 1/dist^2 and we have sq_dist = dist^2,
141 // we can use: weight = 1/sq_dist
142 // This avoids both sqrt() and pow() calls
143 for (size_t i = 0; i < num_found; i++)
144 {
145 double weight = 1.0 / sq_dists[i]; // 1/dist^2 = 1/sq_dist
146 sum_weights += weight;
147 sum_weighted_values += weight * m_values[indices[i]];
148 }
149 }
150 else
151 {
152 // *** GENERAL CASE for arbitrary power ***
153 // Need to calculate actual distance and apply pow()
154 for (size_t i = 0; i < num_found; i++)
155 {
156 double dist = std::sqrt(sq_dists[i]);
157 double weight = 1.0 / std::pow(dist, m_power);
158 sum_weights += weight;
159 sum_weighted_values += weight * m_values[indices[i]];
160 }
161 }
162
163 return sum_weighted_values / sum_weights;
164}

166void SpatialInterpolator::Interpolate(std::vector<Point2D> const& query_points,
167 std::vector<double>& results) const
168{
169 results.resize(query_points.size());
170 size_t const k = std::min((size_t) m_k_neighbors, m_cloud.pts.size());
171
172#ifdef _OPENMP
173 #pragma omp parallel
174 {
175 // Thread-local storage to avoid allocation overhead
176 std::vector<unsigned int> indices(k);
177 std::vector<double> sq_dists(k);
178
179 #pragma omp for schedule(guided, 128) nowait
180 for (size_t i = 0; i < query_points.size(); i++)
181 {
182 double const query_pt[2] = {query_points[i].x, query_points[i].y};
183
184 unsigned int num_found = m_kdtree->knnSearch(query_pt, k,
185 indices.data(),
186 sq_dists.data());
187
188 if (num_found == 0)
189 {
190 results[i] = 0.0; // or m_dMissingValue
191 continue;
192 }
193
194 // Check if we're exactly on a data point
195 if (sq_dists[0] < EPSILON)
196 {
197 results[i] = m_values[indices[0]];
198 continue;
199 }
200
201 // IDW calculation - optimized for power=2.0
202 double sum_weights = 0.0;
203 double sum_weighted_values = 0.0;
204
205 if (m_power == 2.0)
206 {
207 for (size_t j = 0; j < num_found; j++)
208 {
209 double weight = 1.0 / sq_dists[j];
210 sum_weights += weight;
211 sum_weighted_values += weight * m_values[indices[j]];
212 }
213 }
214 else
215 {
216 for (size_t j = 0; j < num_found; j++)
217 {
218 double dist = std::sqrt(sq_dists[j]);
219 double weight = 1.0 / std::pow(dist, m_power);
220 sum_weights += weight;
221 sum_weighted_values += weight * m_values[indices[j]];
222 }
223 }
224
225 results[i] = sum_weighted_values / sum_weights;
226 }
227 }
228#else
229 // Serial fallback - still optimized with reused buffers
230 std::vector<unsigned int> indices(k);
231 std::vector<double> sq_dists(k);
232
233 for (size_t i = 0; i < query_points.size(); i++)
234 {
235 results[i] = Interpolate(query_points[i].x, query_points[i].y);
236 }
237#endif
238}

292void DualSpatialInterpolator::InterpolatePoint(double x, double y,
293 double& result_x, double& result_y,
294 std::vector<unsigned int>& indices,
295 std::vector<double>& sq_dists) const
296{
297 double const query_pt[2] = {x, y};
298 size_t const k = std::min((size_t) m_k_neighbors, m_cloud.pts.size());
299
300 unsigned int num_found = m_kdtree->knnSearch(query_pt, k,
301 indices.data(),
302 sq_dists.data());
303
304 if (num_found == 0)
305 {
306 result_x = result_y = 0.0;
307 return;
308 }
309
310 // Check if we're exactly on a data point
311 if (sq_dists[0] < EPSILON)
312 {
313 result_x = m_values_x[indices[0]];
314 result_y = m_values_y[indices[0]];
315 return;
316 }
317
318 // IDW for both X and Y simultaneously - optimized for power=2.0
319 double sum_weights = 0.0;
320 double sum_weighted_x = 0.0;
321 double sum_weighted_y = 0.0;
322
323 if (m_power == 2.0)
324 {
325 for (size_t i = 0; i < num_found; i++)
326 {
327 double weight = 1.0 / sq_dists[i];
328 sum_weights += weight;
329 sum_weighted_x += weight * m_values_x[indices[i]];
330 sum_weighted_y += weight * m_values_y[indices[i]];
331 }
332 }
333 else
334 {
335 for (size_t i = 0; i < num_found; i++)
336 {
337 double dist = std::sqrt(sq_dists[i]);
338 double weight = 1.0 / std::pow(dist, m_power);
339 sum_weights += weight;
340 sum_weighted_x += weight * m_values_x[indices[i]];
341 sum_weighted_y += weight * m_values_y[indices[i]];
342 }
343 }
344
345 result_x = sum_weighted_x / sum_weights;
346 result_y = sum_weighted_y / sum_weights;
347}
