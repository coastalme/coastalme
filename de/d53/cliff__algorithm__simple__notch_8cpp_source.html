56CCliffResults CCliffAlgorithmSimpleNotch::ProcessTimestep(const CCliffData& cliffData)
57{
58 CCliffResults results;
59
60 // Generate unique key for this cliff point
61 string strKey = GenerateCliffKey(cliffData.nCoast, cliffData.nPointOnCoast);
62
63 // Get current notch depth for this cliff point
64 double dCurrentNotchDepth = 0.0;
65 if (m_mapNotchDepth.find(strKey) != m_mapNotchDepth.end())
66 {
67 dCurrentNotchDepth = m_mapNotchDepth[strKey];
68 }
69
70 // Calculate notch erosion for this timestep
71 double dNotchErosion = CalculateNotchErosion(cliffData.dWaveEnergy, cliffData.dTimeStep);
72
73 // Update notch depth, constrained by maximum depth
74 double dNewNotchDepth = min(dCurrentNotchDepth + dNotchErosion, m_dMaxNotchDepth);
75 m_mapNotchDepth[strKey] = dNewNotchDepth;
76
77 // Update accumulated wave energy
78 double dCurrentEnergy = 0.0;
79 if (m_mapAccumulatedWaveEnergy.find(strKey) != m_mapAccumulatedWaveEnergy.end())
80 {
81 dCurrentEnergy = m_mapAccumulatedWaveEnergy[strKey];
82 }
83 m_mapAccumulatedWaveEnergy[strKey] = dCurrentEnergy + cliffData.dWaveEnergy;
84
85 // Set basic erosion results
86 results.dErosionVolume = dNotchErosion * (cliffData.CliffTop.dGetY() - cliffData.CliffToe.dGetY()); // Approximate volume
87
88 // Check if collapse should occur
89 if (dNewNotchDepth >= m_dNotchDepthAtCollapse)
90 {
91 results.bCollapseOccurred = true;
92
93 // Calculate collapse volumes
94 double dCollapseDepth = dNewNotchDepth;
95 CalculateCollapseVolumes(cliffData, dCollapseDepth,
96 results.dCollapseVolume, results.dCollapseVolumeFine,
97 results.dCollapseVolumeSand, results.dCollapseVolumeCoarse);
98
99 // Reset notch depth after collapse
100 m_mapNotchDepth[strKey] = 0.0;
101
102 // Update cliff positions after collapse (simplified - move toe inland)
103 double dCliffHeight = cliffData.CliffTop.dGetY() - cliffData.CliffToe.dGetY();
104 double dCliffWidth = cliffData.CliffTop.dGetX() - cliffData.CliffToe.dGetX();
105
106 // Calculate retreat based on collapse depth
107 double dRetreatsRatio = dCollapseDepth / dCliffHeight;
108 double dRetreatDistance = dRetreatsRatio * dCliffWidth;
109
110 // Move cliff toe inland
111 results.NewCliffToe.SetX(cliffData.CliffToe.dGetX() + dRetreatDistance);
112 results.NewCliffToe.SetY(cliffData.CliffToe.dGetY());
113
114 // Move cliff top inland proportionally
115 results.NewCliffTop.SetX(cliffData.CliffTop.dGetX() + dRetreatDistance * 0.5);
116 results.NewCliffTop.SetY(cliffData.CliffTop.dGetY());
117 }
118 else
119 {
120 // No collapse - maintain current positions
121 results.NewCliffToe = cliffData.CliffToe;
122 results.NewCliffTop = cliffData.CliffTop;
123 }
124
125 return results;
126}

156double CCliffAlgorithmSimpleNotch::CalculateNotchErosion(double dWaveEnergy, double dTimeStep) const
157{
158 // Simple erosion model: notch erosion = wave energy / resistance
159 // Convert timestep from hours to seconds for energy calculation
160 double dTimeStepSeconds = dTimeStep * 3600.0;
161
162 // Calculate erosion depth based on wave energy and resistance
163 double dErosionDepth = (dWaveEnergy * dTimeStepSeconds) / m_dCliffErosionResistance;
164
165 return max(0.0, dErosionDepth);
166}

169void CCliffAlgorithmSimpleNotch::CalculateCollapseVolumes(const CCliffData& cliffData, double dCollapseDepth,
170 double& dTotalVolume, double& dFineVolume,
171 double& dSandVolume, double& dCoarseVolume) const
172{
173 // Calculate cliff dimensions
174 double dCliffHeight = cliffData.CliffTop.dGetY() - cliffData.CliffToe.dGetY();
175 double dCliffWidth = cliffData.CliffTop.dGetX() - cliffData.CliffToe.dGetX();
176
177 // Calculate total collapse volume (simplified triangular approximation)
178 dTotalVolume = 0.5 * dCollapseDepth * dCliffHeight * dCliffWidth;
179
180 // Simplified sediment size distribution (would normally come from cliff geology)
181 dFineVolume = dTotalVolume * 0.3; // 30% fine sediment
182 dSandVolume = dTotalVolume * 0.5; // 50% sand
183 dCoarseVolume = dTotalVolume * 0.2; // 20% coarse sediment
184}
