73bool CGeomILine::bIsPresent(int const nX, int const nY)
74{
75 int nSize = static_cast<int>(m_VPoints.size());
76 if (nSize == 0)
77 return false;
78
79 for (int n = 0; n < nSize; n++)
80 {
81 if ((nX == m_VPoints[n].nGetX()) && (nY == m_VPoints[n].nGetY()))
82 return true;
83 }
84
85 return false;
86}
