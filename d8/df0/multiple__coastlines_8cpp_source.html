40int CSimulation::nDoMultipleCoastlines(void)
41{
42 // Check all coastlines
43 for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
44 {
45 // Check all profiles
46 for (int n = 0; n < m_VCoast[nCoast].nGetNumProfiles(); n++)
47 {
48 CGeomProfile* pProfile = m_VCoast[nCoast].pGetProfileWithDownCoastSeq(n);
49 int const nProfile = pProfile->nGetProfileID();
50 bool const bCoastStart = pProfile->bStartOfCoast();
51 bool const bCoastEnd = pProfile->bEndOfCoast();
52
53 // Check every cell that is 'under' this profile
54 for (int nCell = 0; nCell < pProfile->nGetNumCellsInProfile(); nCell++)
55 {
56 CGeom2DIPoint const* pCell = pProfile->pPtiGetCellInProfile(nCell);
57 int const nX = pCell->nGetX();
58 int const nY = pCell->nGetY();
59
60 // Have we hit a cell which is 'under' a coast-normal profile belonging to another coast? NOTE Is a problem if get more than two coast normals passing through this cell
61 int nHitProfileCoast = m_pRasterGrid->m_Cell[nX][nY].nGetProfileCoastID();
62 if ((nHitProfileCoast != INT_NODATA) && (nHitProfileCoast != nCoast))
63 {
64 // Yes, we have hit a profile which belongs to a different coast
65 int const nHitProfile = m_pRasterGrid->m_Cell[nX][nY].nGetProfileID();
66
67 // Truncate both profiles
68 int const nRtn = nTruncateProfilesDifferentCoasts(nCoast, nProfile, nCell, nHitProfileCoast, nHitProfile, nX, nY, bCoastStart, bCoastEnd);
69 if (nRtn != RTN_OK)
70 return nRtn;
71 }
72 else
73 {
74 // Now try an adjacent point, does not matter wehich one NOTE Is a problem if get more than two coast normals passing through this cell
75 nHitProfileCoast = m_pRasterGrid->m_Cell[nX][nY+1].nGetProfileCoastID();
76 if ((nHitProfileCoast != INT_NODATA) && (nHitProfileCoast != nCoast))
77 {
78 // Yes, we have hit a profile which belongs to a different coast
79 int const nHitProfile = m_pRasterGrid->m_Cell[nX][nY+1].nGetProfileID();
80
81 // Truncate both profiles
82 int const nRtn = nTruncateProfilesDifferentCoasts(nCoast, nProfile, nCell, nHitProfileCoast, nHitProfile, nX, nY+1, bCoastStart, bCoastEnd);
83 if (nRtn != RTN_OK)
84 return nRtn;
85 }
86 }
87
88 // Have we hit a cell which is 'under' another coast?
89 if (m_pRasterGrid->m_Cell[nX][nY].bIsCoastline())
90 {
91 // Yes this is a coastline cell, as well as a coast-normal profile cell
92 int const nHitCoast = m_pRasterGrid->m_Cell[nX][nY].nGetCoastline();
93 if (nHitCoast != nCoast)
94 {
95 // We have hit a different coastline, so truncate this profile
96 int const nRtn = nTruncateProfileHitDifferentCoast(nCoast, nProfile, nCell, nX, nY, bCoastStart, bCoastEnd);
97 if (nRtn != RTN_OK)
98 return nRtn;
99 }
100 }
101 else
102 {
103 // We also need to check an adjacent cell, doesn't matter which one
104 if (m_pRasterGrid->m_Cell[nX][nY+1].bIsCoastline())
105 {
106 // Yes this is a coastline cell, as well as a coast-normal profile cell
107 int const nHitCoast = m_pRasterGrid->m_Cell[nX][nY+1].nGetCoastline();
108 if (nHitCoast != nCoast)
109 {
110 // We have hit a different coastline, so truncate this profile
111 int const nRtn = nTruncateProfileHitDifferentCoast(nCoast, nProfile, nCell, nX, nY, bCoastStart, bCoastEnd);
112 if (nRtn != RTN_OK)
113 return nRtn;
114 }
115 }
116 }
117 }
118 }
119 }
120
121 return RTN_OK;
122}

127int CSimulation::nTruncateProfilesDifferentCoasts(int const nCoast, int const nProfile, int nCell, int const nHitProfileCoast, int const nHitProfile, int nX, int nY, bool const bStartCoastEdgeProfile, bool const bEndCoastEdgeProfile)
128{
129 if ((nProfile == INT_NODATA) || (nHitProfile == INT_NODATA))
130 {
131 // Should never happen
132 return RTN_ERR_CELL_MARKED_PROFILE_COAST_BUT_NOT_PROFILE;
133 }
134
135 // OK, get pointers to 'this' profile and to the hit profile
136 CGeomProfile* pProfile = m_VCoast[nCoast].pGetProfile(nProfile);
137 CGeomProfile* pHitProfile = m_VCoast[nHitProfileCoast].pGetProfile(nHitProfile);
138
139 // Get details of the cells in this profile
140 vector<CGeom2DIPoint>* pVProfileCells = pProfile->pPtiVGetCellsInProfile();
141 int const nProfileLen = static_cast<int>(pVProfileCells->size());
142
143 // Are either of the profiles grid-edge profiles?
144 bool bProfileGridEdge = false;
145 bool bHitProfileGridEdge = false;
146 if (pProfile->bStartOfCoast() || pProfile->bEndOfCoast())
147 bProfileGridEdge = true;
148 if (pHitProfile->bStartOfCoast() || pHitProfile->bEndOfCoast())
149 bHitProfileGridEdge = true;
150
151 // Are both profiles grid-edge profiles?
152 if (bProfileGridEdge && bHitProfileGridEdge)
153 {
154 // Yes, so we need to treat these profiles differently. Get the start point of each profile
155 CGeom2DIPoint const* pPtiProfileStart = pProfile->pPtiGetStartPoint();
156 CGeom2DIPoint const* pPtiHitProfileStart = pHitProfile->pPtiGetStartPoint();
157
158 // And get the distance between these
159 double const dDist = dGetDistanceBetween(pPtiProfileStart, pPtiHitProfileStart);
160 nCell = static_cast<int>(dDist - GAP_BETWEEN_DIFFERENT_COAST_PROFILES);
161
162 // Safety check
163 nCell = tMax(nCell, 0);
164
165 // Finally get the grid CRS location of the new profile endpoint
166 nX = pVProfileCells->at(nCell).nGetX();
167 nY = pVProfileCells->at(nCell).nGetY();
168 }
169
170 // Calculate the truncated length of the list of cells in 'this' profile
171 int const nProfileNewLen = tMax(nCell - GAP_BETWEEN_DIFFERENT_COAST_PROFILES, MIN_PROFILE_SIZE);
172
173 // Next unmark the cells that will no longer be 'under' the profile
174 for (int nn = nProfileNewLen-1; nn < nProfileLen; nn++)
175 {
176 int const nXThis = pVProfileCells->at(nn).nGetX();
177 int const nYThis = pVProfileCells->at(nn).nGetY();
178
179 if ((m_pRasterGrid->m_Cell[nXThis][nYThis].nGetProfileID() == nProfile) && (m_pRasterGrid->m_Cell[nXThis][nYThis].nGetProfileCoastID() == nCoast))
180 m_pRasterGrid->m_Cell[nXThis][nYThis].SetCoastAndProfileID(INT_NODATA, INT_NODATA);
181 }
182
183 // And truncate the list of cells in the profile
184 pVProfileCells->resize(nProfileNewLen);
185 pProfile->SetCellsInProfile(pVProfileCells);
186
187 // Set the profile's end point (grid CRS)
188 CGeom2DIPoint PtiLast = pVProfileCells->back();
189 pProfile->SetEndPoint(&PtiLast);
190
191 // Now truncate the profile's CGeomMultiLine (external CRS)
192 int nRtn = nTruncateProfileMultiLineDifferentCoasts(pProfile, nX, nY);
193 if (nRtn != RTN_OK)
194 return nRtn;
195
196 // And flag as truncated
197 pProfile->SetTruncatedDifferentCoast(true);
198
199 if (m_nLogFileDetail >= LOG_FILE_ALL)
200 {
201 string strTmp;
202 if (bStartCoastEdgeProfile || bEndCoastEdgeProfile)
203 {
204 strTmp += " grid-edge ";
205
206 if (bStartCoastEdgeProfile)
207 strTmp += "start";
208
209 if (bEndCoastEdgeProfile)
210 strTmp += "end";
211 }
212
213 LogStream << m_ulIter << ": coast " << nCoast << strTmp << " profile " << nProfile << " hit profile belonging to another coast at [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}. Profile truncated, length of profile " << nProfile << " was " << nProfileLen << " cells, is now " << nProfileNewLen << " cells." << endl;
214 }
215
216 // Get details of the cells in the hit profile
217 vector<CGeom2DIPoint>* pVHitProfileCells = pHitProfile->pPtiVGetCellsInProfile();
218 int const nHitProfileLen = static_cast<int>(pVHitProfileCells->size());
219
220 int const nHitCell = pHitProfile->nGetIndexOfCellInProfile(nX, nY);
221 if (nHitCell == INT_NODATA)
222 return RTN_ERR_CELL_NOT_FOUND_IN_HIT_PROFILE_DIFFERENT_COASTS;
223
224 // Calculate the truncated length of the list of cells in the hit profile
225 int const nHitProfileNewLen = tMax(nHitCell - GAP_BETWEEN_DIFFERENT_COAST_PROFILES, MIN_PROFILE_SIZE);
226
227 // Next unmark the cells that will no longer be 'under' the hit profile
228 for (int nn = nHitProfileNewLen-1; nn < nHitProfileLen; nn++)
229 {
230 int const nXThis = pVHitProfileCells->at(nn).nGetX();
231 int const nYThis = pVHitProfileCells->at(nn).nGetY();
232
233 if ((m_pRasterGrid->m_Cell[nXThis][nYThis].nGetProfileID() == nHitProfile) && (m_pRasterGrid->m_Cell[nXThis][nYThis].nGetProfileCoastID() == nHitProfileCoast))
234 m_pRasterGrid->m_Cell[nXThis][nYThis].SetCoastAndProfileID(INT_NODATA, INT_NODATA);
235 }
236
237 // And truncate the hit profile
238 pVHitProfileCells->resize(nHitProfileNewLen);
239 pHitProfile->SetCellsInProfile(pVHitProfileCells);
240
241 // Set the hit profile's end point (grid CRS)
242 PtiLast = pVHitProfileCells->back();
243 pHitProfile->SetEndPoint(&PtiLast);
244
245 // Now truncate the profile's CGeomMultiLine (external CRS)
246 nRtn = nTruncateProfileMultiLineDifferentCoasts(pProfile, nX, nY);

257 {
258 strTmp += " grid-edge ";
259
260 if (bStartCoastEdgeProfile)
261 strTmp += "start";
262
263 if (bEndCoastEdgeProfile)
264 strTmp += "end";
265 }
266
267 LogStream << m_ulIter << ": coast " << nHitProfileCoast << strTmp << " profile " << nHitProfile << " also truncated, length of profile " << nHitProfile << " was " << nHitProfileLen << " cells, is now " << nHitProfileNewLen << " cells." << endl;
268 }
269
270 // // DEBUG CODE ================
271 // m_nGISSave++;
272 // if (! bWriteVectorGISFile(VECTOR_PLOT_COAST, &VECTOR_PLOT_COAST_TITLE))
273 // return false;

284 return RTN_OK;

290int CSimulation::nTruncateProfileHitDifferentCoast(int const nCoast, int const nProfile, int const nCell, int const nX, int const nY, bool const bStartCoastEdgeProfile, bool const bEndCoastEdgeProfile)
291{
292 // OK, get a pointer to 'this' profile
293 CGeomProfile* pProfile = m_VCoast[nCoast].pGetProfile(nProfile);
294
295 // Now get details of the cells in this profile
296 vector<CGeom2DIPoint>* pVProfileCells = pProfile->pPtiVGetCellsInProfile();
297 int nProfileLen = static_cast<int>(pVProfileCells->size());
298
299 // Calculate the truncated length of the list of cells in 'this' profile
300 int nProfileNewLen = tMax(nCell - GAP_BETWEEN_DIFFERENT_COAST_PROFILES, MIN_PROFILE_SIZE);
301
302 // Get the lengths of the adjacent, this-coast, profiles
303 CGeomProfile const* pUpCoastProfile = pProfile->pGetUpCoastAdjacentProfile();
304 CGeomProfile const* pDownCoastProfile = pProfile->pGetDownCoastAdjacentProfile();
305 int nUpCoastProfileLen = INT_NODATA;
306 int nDownCoastProfileLen = INT_NODATA;
307 if (pUpCoastProfile != NULL)
308 nUpCoastProfileLen = pUpCoastProfile->nGetNumCellsInProfile();
309 if (pDownCoastProfile != NULL)
310 nDownCoastProfileLen = pDownCoastProfile->nGetNumCellsInProfile();
311
312 // And calculate the average length of adjacent profiles
313 int nAvgAdjacentProfileLen;
314 if (pUpCoastProfile == NULL)
315 nAvgAdjacentProfileLen = nDownCoastProfileLen;
316 else if (pDownCoastProfile == NULL)
317 nAvgAdjacentProfileLen = nUpCoastProfileLen;
318 else
319 nAvgAdjacentProfileLen = (nUpCoastProfileLen + nDownCoastProfileLen) / 2;
320
321 // Use the average length of adjacent profiles to further truncated this profile, if necessary
322 nProfileNewLen = tMin(nProfileNewLen, nAvgAdjacentProfileLen);
323
324 // We have the new profile length so next unmark the cells that will no longer be 'under' the profile
325 for (int nn = nProfileNewLen-1; nn < nProfileLen; nn++)
326 {
327 int const nXThis = pVProfileCells->at(nn).nGetX();
328 int const nYThis = pVProfileCells->at(nn).nGetY();
329
330 if ((m_pRasterGrid->m_Cell[nXThis][nYThis].nGetProfileID() == nProfile) && (m_pRasterGrid->m_Cell[nXThis][nYThis].nGetProfileCoastID() == nCoast))
331 m_pRasterGrid->m_Cell[nXThis][nYThis].SetCoastAndProfileID(INT_NODATA, INT_NODATA);
332 }
333
334 // And truncate the list of cells in the profile
335 pVProfileCells->resize(nProfileNewLen);
336 pProfile->SetCellsInProfile(pVProfileCells);
337
338 // Set the profile's end point (grid CRS)
339 CGeom2DIPoint const PtiLast = pVProfileCells->back();
340 pProfile->SetEndPoint(&PtiLast);
341
342 // Now truncate the profile's CGeomMultiLine (external CRS)
343 int const nRtn = nTruncateProfileMultiLineDifferentCoasts(pProfile, nX, nY);
344 if (nRtn != RTN_OK)
345 return nRtn;
346
347 // And flag as truncated
348 pProfile->SetTruncatedDifferentCoast(true);
349
350 if (m_nLogFileDetail >= LOG_FILE_ALL)
351 {
352 string strTmp;
353 if (bStartCoastEdgeProfile || bEndCoastEdgeProfile)
354 {
355 strTmp += " grid-edge ";
356
357 if (bStartCoastEdgeProfile)
358 strTmp += "start";
359
360 if (bEndCoastEdgeProfile)
361 strTmp += "end";
362 }
363
364 LogStream << m_ulIter << ": coast " << nCoast << strTmp << " profile " << nProfile << " hit another coast at [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}. Profile truncated, length of profile " << nProfile << " was " << nProfileLen << " cells, is now " << nProfileNewLen << " cells." << endl;
365 }
366
367
368 // DEBUG CODE ================
369 m_nGISSave++;
370 if (! bWriteVectorGISFile(VECTOR_PLOT_COAST, &VECTOR_PLOT_COAST_TITLE))
371 return false;
372 if (! bWriteVectorGISFile(VECTOR_PLOT_NORMALS, &VECTOR_PLOT_NORMALS_TITLE))
373 return false;
374 if (! bWriteVectorGISFile(VECTOR_PLOT_INVALID_NORMALS, &VECTOR_PLOT_INVALID_NORMALS_TITLE))
375 return false;
376 if (! bWriteRasterGISFile(RASTER_PLOT_NORMAL_PROFILE, &RASTER_PLOT_NORMAL_PROFILE_TITLE))
377 return false;
378 if (! bWriteRasterGISFile(RASTER_PLOT_POLYGON, &RASTER_PLOT_POLYGON_TITLE))
379 return false;
380 // DEBUG CODE ================
381
382 return RTN_OK;
383}

388int CSimulation::nTruncateProfileMultiLineDifferentCoasts(CGeomProfile* pProfile, int const nX, int const nY)
389{
390 // Find which multiline line segment 'contains' this extCRS end point, then truncate at this point
391 bool bFound = false;
392
393 int const nProfileLineSegments = pProfile->CGeomMultiLine::nGetNumLineSegments();
394 for (int nSeg = 0; nSeg < nProfileLineSegments; nSeg++)
395 {
396 // Search each segment
397 int const nNumCoinc = pProfile->CGeomMultiLine::nGetNumCoincidentProfilesInLineSegment(nSeg);
398 for (int nCoinc = 0; nCoinc < nNumCoinc; nCoinc++)
399 {
400 vector<CGeom2DPoint>& pVPt = pProfile->CGeomMultiLine::pGetPoints();
401
402 for (int nLin = 0; nLin < static_cast<int>(pVPt.size())-1; nLin++)
403 {
404 double const dX1 = pVPt[nLin].dGetX();
405 double const dY1 = pVPt[nLin].dGetY();
406 double const dX2 = pVPt[nLin+1].dGetX();
407 double const dY2 = pVPt[nLin+1].dGetY();
408
409 double const dXMin = tMin(dX1, dX2);
410 double const dXMax = tMax(dX1, dX2);
411 double const dYMin = tMin(dY1, dY2);
412 double const dYMax = tMax(dY1, dY2);
413
414 double const dX = dGridXToExtCRSX(nX);
415 double const dY = dGridYToExtCRSY(nY);
416
417 if ((dX >= dXMin) && (dX <= dXMax) && (dY >= dYMin) && (dY <= dYMax))
418 {
419 bFound = true;
420
421 pProfile->TruncateProfile(nSeg+1);
422 pVPt.push_back(CGeom2DPoint(dX, dY));
423 pProfile->CGeomMultiLine::SetPoints(pVPt);
424
425 break;
426 }
427 }
428 if (bFound)
429 break;
430 }
431 if (bFound)
432 break;
433 }
434
435 if (! bFound)
436 return RTN_ERR_POINT_NOT_FOUND_IN_MULTILINE_DIFFERENT_COASTS;
437
438 return RTN_OK;
