57void CRWCellLandform::SetLFSubCategory(int const nClassIn)
58{
59 m_nSubCategory = nClassIn;
60
61 if ((nClassIn == LF_SUBCAT_CLIFF_ON_COASTLINE) || (nClassIn == LF_SUBCAT_CLIFF_INLAND))
62 m_nCategory = LF_CAT_CLIFF;
63 else if ((nClassIn == LF_SUBCAT_DRIFT_TALUS) || (nClassIn == LF_SUBCAT_DRIFT_BEACH) || (nClassIn == LF_SUBCAT_DRIFT_MIXED))
64 m_nCategory = LF_CAT_DRIFT;
65 else if ((nClassIn == LF_SUBCAT_SEDIMENT_INPUT_UNCONSOLIDATED) || (nClassIn == LF_SUBCAT_SEDIMENT_INPUT_CONSOLIDATED))
66 m_nCategory = LF_CAT_SEDIMENT_INPUT;
67}
