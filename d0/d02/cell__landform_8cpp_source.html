56void CRWCellLandform::SetLFSubCategory(int const nClassIn)
57{
58 m_nSubCategory = nClassIn;
59
60 if ((nClassIn == LF_SUBCAT_CLIFF_ON_COASTLINE) || (nClassIn == LF_SUBCAT_CLIFF_INLAND))
61 m_nCategory = LF_CAT_CLIFF;
62
63 else if ((nClassIn == LF_SUBCAT_DRIFT_TALUS) || (nClassIn == LF_SUBCAT_DRIFT_BEACH) || (nClassIn == LF_SUBCAT_DRIFT_MIXED))
64 m_nCategory = LF_CAT_DRIFT;
65}
