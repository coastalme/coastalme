27int CSimulation::nCalcExternalForcing(void)
28{
29 // Increment SWL (note that increment may be zero)
30 m_dAccumulatedSeaLevelChange += m_dDeltaSWLPerTimestep;
31
32 // This-iteration mean SWL includes only long-term SWL change
33 m_dThisIterMeanSWL = m_dInitialMeanSWL + m_dAccumulatedSeaLevelChange;
34
35 int const nSize = static_cast<int>(m_VdTideData.size());
36 if (nSize == 0)
37 {
38 // No tide data
39 m_dThisIterSWL = m_dThisIterMeanSWL;
40 }
41 else
42 {
43 // We have tide data
44 static int snTideDataCount = 0;
45
46 // Wrap the tide data, i.e. start again with the first record if we do not have enough
47 if (snTideDataCount > nSize - 1)
48 snTideDataCount = 0;
49
50 // This-iteration SWL includes both tidal change and long-term SWL change
51 m_dThisIterSWL = m_dInitialMeanSWL + m_VdTideData[snTideDataCount];
52
53 snTideDataCount++;
54 }
55
56 m_bHighestSWLSoFar = false;
57 m_bLowestSWLSoFar = false;
58
59 // Maybe update min and max still water levels so far during this simulation
60 if (m_dThisIterSWL > m_dMaxSWLSoFar)
61 {
62 m_bHighestSWLSoFar = true;
63 m_dMaxSWLSoFar = m_dThisIterSWL;
64 }
65
66 if (m_dThisIterSWL < m_dMinSWLSoFar)
67 {
68 m_bLowestSWLSoFar = true;
69 m_dMinSWLSoFar = m_dThisIterSWL;
70 }
71
72 // Update the wave height, orientation and period for this time step and start again with the first record if we do not have enough
73 if (m_bHaveWaveStationData)
74 {
75 static int snWaveStationDataCount = 0;
76
77 if (snWaveStationDataCount > m_nDeepWaterWaveDataNumTimeSteps - 1)
78 {
79 // Wrap the tide data, i.e. start again with the first record if we do not have enough
80 snWaveStationDataCount = 0;
81 }
82
83 // Do we have just a single wave station?
84 if (m_bSingleDeepWaterWaveValues)
85 {
86 // Yes, just a single wave station
87 m_dAllCellsDeepWaterWaveHeight = m_VdTSDeepWaterWaveStationHeight[snWaveStationDataCount];
88 m_dAllCellsDeepWaterWaveAngle = m_VdTSDeepWaterWaveStationAngle[snWaveStationDataCount];
89 m_dAllCellsDeepWaterWavePeriod = m_VdTSDeepWaterWaveStationPeriod[snWaveStationDataCount];
90 }
91 else
92 {
93 // More than one wave station, so update this time step's deep water wave values for use in the nInterpolateAllDeepWaterWaveValues() routine. Note that the order on the vector is determined by the points ID i.e. to ensure that stations match with time series
94 int const nNumberDeepWaterWaveStations = static_cast<int>(m_VnDeepWaterWaveStationID.size());
95 int const nTot = nNumberDeepWaterWaveStations * snWaveStationDataCount;
96
97 for (int j = 0; j < nNumberDeepWaterWaveStations; j++)
98 {
99 m_VdThisIterDeepWaterWaveStationHeight[j] = m_VdTSDeepWaterWaveStationHeight[(m_VnDeepWaterWaveStationID[j] - 1) + nTot];
100 m_VdThisIterDeepWaterWaveStationAngle[j] = m_VdTSDeepWaterWaveStationAngle[(m_VnDeepWaterWaveStationID[j] - 1) + nTot];
101 m_VdThisIterDeepWaterWaveStationPeriod[j] = m_VdTSDeepWaterWaveStationPeriod[(m_VnDeepWaterWaveStationID[j] - 1) + nTot];
102 }
103 }
104
105 snWaveStationDataCount++;
106 }
107
108 return RTN_OK;
109}
