36int CSimulation::nCalcExternalForcing(void)
37{
38 // Increment SWL (note that increment may be zero)
39 m_dAccumulatedSeaLevelChange += m_dDeltaSWLPerTimestep;
40
41 int nSize = static_cast<int>(m_VdTideData.size());
42 if (nSize != 0)
43 {
44 // We have tide data
45 static int snTideDataCount = 0;
46
47 // Wrap the tide data, i.e. start again with the first record if we do not have enough
48 if (snTideDataCount > nSize-1)
49 snTideDataCount = 0;
50
51 // This-iteration SWL includes both tidal change and long-term SWL change
52 m_dThisIterSWL = m_dOrigSWL + m_VdTideData[snTideDataCount] + m_dAccumulatedSeaLevelChange;
53
54 // This-iteration mean SWL includes only long-term SWL change
55 m_dThisIterMeanSWL = m_dOrigSWL + m_dAccumulatedSeaLevelChange;
56
57 // cout << m_dThisIterSWL << endl;
58 snTideDataCount++;
59 }
60
61 // Update min and max still water levels
62 m_dMaxSWL = tMax(m_dThisIterSWL, m_dMaxSWL);
63 m_dMinSWL = tMin(m_dThisIterSWL, m_dMinSWL);
64
65 // Update the wave height, orientation and period for this time step and start again with the first record if we do not have enough
66 if (m_bHaveWaveStationData)
67 {
68 static int snWaveStationDataCount = 0;
69
70 if (snWaveStationDataCount > m_nDeepWaterWaveDataNumTimeSteps-1)
71 {
72 // Wrap the tide data, i.e. start again with the first record if we do not have enough
73 snWaveStationDataCount = 0;
74 }
75
76 // Do we have just a single wave station?
77 if (m_bSingleDeepWaterWaveValues)
78 {
79 // Yes, just a single wave station
80 m_dAllCellsDeepWaterWaveHeight = m_VdTSDeepWaterWaveStationHeight[snWaveStationDataCount];
81 m_dAllCellsDeepWaterWaveAngle = m_VdTSDeepWaterWaveStationAngle[snWaveStationDataCount];
82 m_dAllCellsDeepWaterWavePeriod = m_VdTSDeepWaterWaveStationPeriod[snWaveStationDataCount];
83 }
84 else
85 {
86 // More than one wave station, so update this time step's deep water wave values for use in the nInterpolateAllDeepWaterWaveValues() routine. Note that the order on the vector is determined by the points ID i.e. to ensure that stations match with time series
87 int nNumberDeepWaterWaveStations = static_cast<int>(m_VnDeepWaterWaveStationID.size());
88 int nTot = nNumberDeepWaterWaveStations * snWaveStationDataCount;
89
90 for (int j = 0; j < nNumberDeepWaterWaveStations; j++)
91 {
92 m_VdThisIterDeepWaterWaveStationHeight[j] = m_VdTSDeepWaterWaveStationHeight[(m_VnDeepWaterWaveStationID[j]-1) + nTot];
93 m_VdThisIterDeepWaterWaveStationAngle[j] = m_VdTSDeepWaterWaveStationAngle[(m_VnDeepWaterWaveStationID[j]-1) + nTot];
94 m_VdThisIterDeepWaterWaveStationPeriod[j] = m_VdTSDeepWaterWaveStationPeriod[(m_VnDeepWaterWaveStationID[j]-1) + nTot];
95 }
96 }
97
98 snWaveStationDataCount++;
99 }
100
101 return RTN_OK;
102}
