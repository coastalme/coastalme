51int CSimulation::nCheckInterventionFailures(void)
52{
53 int nInterventionsRemoved = 0;
54 vector<CGeom2DIPoint> vFailedInterventions;
55
56 // Use OpenMP to parallelize the grid scan for intervention failures
57 // Each thread collects failures in a private vector, then we merge them
58 vector<vector<CGeom2DIPoint>> vThreadFailures;
59
60#ifdef _OPENMP
61 int nThreads = omp_get_max_threads();
62 vThreadFailures.resize(nThreads);
63
64 #pragma omp parallel
65 {
66 int nThreadID = omp_get_thread_num();
67 vector<CGeom2DIPoint>& vLocalFailures = vThreadFailures[nThreadID];
68
69 // Parallelize the outer loop with dynamic scheduling for load balancing
70 #pragma omp for schedule(dynamic, 10)
71 for (int nX = 0; nX < m_nXGridSize; nX++)
72 {
73 for (int nY = 0; nY < m_nYGridSize; nY++)
74 {
75 // Check if this cell is an intervention and if its influence zone has failed
76 if (bIsInterventionCell(nX, nY) && bCheckInfluenceZoneFailure(nX, nY))
77 {
78 vLocalFailures.push_back(CGeom2DIPoint(nX, nY));
79 }
80 }
81 }
82 }
83
84 // Merge results from all threads
85 for (const auto& vThreadResult : vThreadFailures)
86 {
87 vFailedInterventions.insert(vFailedInterventions.end(),
88 vThreadResult.begin(), vThreadResult.end());
89 }
90#else
91 // Fallback for non-OpenMP builds
92 for (int nX = 0; nX < m_nXGridSize; nX++)
93 {
94 for (int nY = 0; nY < m_nYGridSize; nY++)
95 {
96 // Check if this cell is an intervention and if its influence zone has failed
97 if (bIsInterventionCell(nX, nY) && bCheckInfluenceZoneFailure(nX, nY))
98 {
99 vFailedInterventions.push_back(CGeom2DIPoint(nX, nY));
100 }
101 }
102 }
103#endif
104
105 // Group contiguous failed interventions and remove them as units
106 vector<bool> vProcessed(vFailedInterventions.size(), false);
107
108 for (size_t i = 0; i < vFailedInterventions.size(); i++)
109 {
110 if (vProcessed[i])
111 continue;
112
113 // Start a new intervention group
114 vector<CGeom2DIPoint> vInterventionGroup;
115 vector<size_t> vToProcess;
116 vToProcess.push_back(i);
117
118 // Find all contiguous intervention cells in this group
119 while (!vToProcess.empty())
120 {
121 size_t nCurrent = vToProcess.back();
122 vToProcess.pop_back();
123
124 if (vProcessed[nCurrent])
125 continue;
126
127 vProcessed[nCurrent] = true;
128 vInterventionGroup.push_back(vFailedInterventions[nCurrent]);
129
130 int nX = vFailedInterventions[nCurrent].nGetX();
131 int nY = vFailedInterventions[nCurrent].nGetY();
132
133 // Check neighboring cells for more intervention failures (8-direction)
134 int nXInc[8] = {-1, -1, -1, 0, 0, 1, 1, 1};
135 int nYInc[8] = {-1, 0, 1, -1, 1, -1, 0, 1};
136
137 for (int nDirection = 0; nDirection < 8; nDirection++)
138 {
139 int nXAdj = nX + nXInc[nDirection];
140 int nYAdj = nY + nYInc[nDirection];
141
142 if (!bIsWithinValidGrid(nXAdj, nYAdj))
143 continue;
144
145 // Look for this adjacent cell in our failed interventions list
146 for (size_t j = 0; j < vFailedInterventions.size(); j++)
147 {
148 if (!vProcessed[j] &&
149 vFailedInterventions[j].nGetX() == nXAdj &&
150 vFailedInterventions[j].nGetY() == nYAdj)
151 {
152 vToProcess.push_back(j);
153 break;
154 }
155 }
156 }
157 }
158
159 // Remove this entire intervention group
160 if (!vInterventionGroup.empty())
161 {
162 for (const auto& point : vInterventionGroup)
163 {
164 m_pRasterGrid->m_Cell[point.nGetX()][point.nGetY()].RemoveIntervention();
165 }
166 nInterventionsRemoved++;
167
168 LogStream << m_ulIter << ": intervention group of " << vInterventionGroup.size()
169 << " cells failed due to influence zone undermining and was removed" << endl;
170 }
171 }
172
173 if (nInterventionsRemoved > 0)
174 {
175 LogStream << m_ulIter << ": total of " << nInterventionsRemoved
176 << " intervention groups failed due to influence zone undermining" << endl;
177 }
178
179 return RTN_OK;
180}

185void CSimulation::SetInterventionTriggerDepths(double const dTriggerDepth)
186{
187 int nInterventionsWithTriggers = 0;
188
189 for (int nX = 0; nX < m_nXGridSize; nX++)
190 {
191 for (int nY = 0; nY < m_nYGridSize; nY++)
192 {
193 if (bIsInterventionCell(nX, nY))
194 {
195 m_pRasterGrid->m_Cell[nX][nY].SetInterventionTriggerDepth(dTriggerDepth);
196 nInterventionsWithTriggers++;
197 }
198 }
199 }
200
201 LogStream << "Set trigger depth of " << dTriggerDepth << " m for "
202 << nInterventionsWithTriggers << " intervention cells" << endl;
203}

208bool CSimulation::bCheckInfluenceZoneFailure(int const nIntX, int const nIntY) const
209{
210 // Get the trigger elevation for this intervention cell
211 double dTriggerElev = m_pRasterGrid->m_Cell[nIntX][nIntY].dGetInterventionTriggerElev();
212
213 // If no valid trigger elevation is set, no failure
214 if (bFPIsEqual(dTriggerElev, DBL_NODATA, TOLERANCE))
215 return false;
216
217 // Calculate search radius in grid cells (convert from meters to cells)
218 int nSearchRadius = static_cast<int>(ceil(m_dInterventionInfluenceDistance / m_dCellSide));
219
220 // Pre-calculate squared influence distance to avoid sqrt in inner loop
221 double dInfluenceDistanceSquared = m_dInterventionInfluenceDistance * m_dInterventionInfluenceDistance;
222 double dCellSideSquared = m_dCellSide * m_dCellSide;
223
224 bool bFailureFound = false;
225
226 // Parallelize the influence zone search for large zones (radius > 3)
227 // For small zones, the overhead isn't worth it
228#ifdef _OPENMP
229 if (nSearchRadius > 3)
230 {
231 #pragma omp parallel for collapse(2) shared(bFailureFound) if(!bFailureFound)
232 for (int nXOffset = -nSearchRadius; nXOffset <= nSearchRadius; nXOffset++)
233 {
234 for (int nYOffset = -nSearchRadius; nYOffset <= nSearchRadius; nYOffset++)
235 {
236 // Skip if failure already found by another thread
237 if (bFailureFound)
238 continue;
239
240 int nCheckX = nIntX + nXOffset;
241 int nCheckY = nIntY + nYOffset;
242
243 // Skip if outside grid bounds
244 if (!bIsWithinValidGrid(nCheckX, nCheckY))
245 continue;
246
247 // Fast distance check using squared distances (avoid sqrt)
248 double dDistanceSquared = (nXOffset * nXOffset + nYOffset * nYOffset) * dCellSideSquared;
249
250 // Skip if outside influence distance
251 if (dDistanceSquared > dInfluenceDistanceSquared)
252 continue;
253
254 // Check if this cell's ground elevation is below the trigger level
255 double dCurrentElev = m_pRasterGrid->m_Cell[nCheckX][nCheckY].dGetSedimentTopElev();
256
257 if (dCurrentElev < dTriggerElev)
258 {
259 #pragma omp critical
260 {
261 bFailureFound = true; // Failure detected in influence zone
262 }
263 }
264 }
265 }
266 }
267 else
268#endif
269 {
270 // Serial version for small influence zones or non-OpenMP builds
271 for (int nXOffset = -nSearchRadius; nXOffset <= nSearchRadius; nXOffset++)
272 {
273 for (int nYOffset = -nSearchRadius; nYOffset <= nSearchRadius; nYOffset++)
274 {
275 int nCheckX = nIntX + nXOffset;
276 int nCheckY = nIntY + nYOffset;
277
278 // Skip if outside grid bounds
279 if (!bIsWithinValidGrid(nCheckX, nCheckY))
280 continue;
281
282 // Fast distance check using squared distances (avoid sqrt)
283 double dDistanceSquared = (nXOffset * nXOffset + nYOffset * nYOffset) * dCellSideSquared;
284
285 // Skip if outside influence distance
286 if (dDistanceSquared > dInfluenceDistanceSquared)
287 continue;
288
289 // Check if this cell's ground elevation is below the trigger level
290 double dCurrentElev = m_pRasterGrid->m_Cell[nCheckX][nCheckY].dGetSedimentTopElev();
291
292 if (dCurrentElev < dTriggerElev)
293 {
294 return true; // Failure detected in influence zone
295 }
296 }
297 }
298 }
299
300 return bFailureFound;
301}
