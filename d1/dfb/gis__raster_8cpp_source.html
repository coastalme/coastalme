63void CSimulation::InitializeGDALPerformance(void)
64{
65 // Configure GDAL for optimal performance
66 // Enable GDAL threading - use all available CPU cores
67#ifdef _OPENMP
68 CPLSetConfigOption("GDAL_NUM_THREADS", "ALL_CPUS");
69#else
70 CPLSetConfigOption("GDAL_NUM_THREADS", "2"); // Fallback for non-OpenMP builds
71#endif
72
73 // Optimize GDAL memory usage and caching
74 CPLSetConfigOption("GDAL_CACHEMAX", "2GB"); // 2GB cache for large grids (was 1GB)
75 CPLSetConfigOption("GDAL_DISABLE_READDIR_ON_OPEN", "TRUE"); // Faster file access
76 CPLSetConfigOption("VSI_CACHE", "TRUE"); // Enable virtual file system cache
77 CPLSetConfigOption("VSI_CACHE_SIZE", "256MB"); // 256MB VSI cache
78
79 // Optimize grid creation performance
80 CPLSetConfigOption("GDAL_GRID_MAX_POINTS_PER_QUADTREE_LEAF", "512"); // Faster spatial indexing
81
82 // Disable GDAL warnings for cleaner output (optional)
83 // CPLSetConfigOption("CPL_LOG", "/dev/null");
84
85 m_bGDALOptimisations = true;
86}

91int CSimulation::nReadRasterBasementDEM(void)
92{
93 // Initialize GDAL performance settings (only needs to be done once)
94 static bool bGDALInitialized = false;
95
96 if (! bGDALInitialized)
97 {
98 InitializeGDALPerformance();
99 bGDALInitialized = true;
100 }
101
102 // Use GDAL to create a dataset object, which then opens the DEM file
103 GDALDataset *pGDALDataset = static_cast<GDALDataset*>(GDALOpen(m_strInitialBasementDEMFile.c_str(), GA_ReadOnly));
104
105 if (NULL == pGDALDataset)
106 {
107 // Can't open file (note will already have sent GDAL error message to stdout)
108 cerr << ERR << "cannot open " << m_strInitialBasementDEMFile << " for input: " << CPLGetLastErrorMsg() << endl;
109 return RTN_ERR_DEMFILE;
110 }
111
112 // Opened OK, so get GDAL basement DEM dataset information
113 m_strGDALBasementDEMDriverCode = pGDALDataset->GetDriver()->GetDescription();
114 m_strGDALBasementDEMDriverDesc = pGDALDataset->GetDriver()->GetMetadataItem(GDAL_DMD_LONGNAME);
115 m_strGDALBasementDEMProjection = pGDALDataset->GetProjectionRef();
116
117 if (m_strGDALBasementDEMProjection.empty())
118 {
119 // TODO
120 m_strGDALBasementDEMProjection = "";
121 // pGDALDataset->SetProjectionRef(m_strGDALBasementDEMProjection);
122
123// ENGCRS["Plane",
124// EDATUM["Unknown engineering datum"],
125// CS[Cartesian,2],
126// AXIS["(E)",east,
127// ORDER[1],
128// LENGTHUNIT["Meter",1]],
129// AXIS["(N)",north,
130// ORDER[2],
131// LENGTHUNIT["Meter",1]]]
132 }
133 else
134 {
135 // We have reference units, so check that they are in metres (note US spelling)
136 if (! m_strGDALBasementDEMProjection.empty())
137 {
138 string const strTmp = strToLower(&m_strGDALBasementDEMProjection);
139
140 if ((strTmp.find("meter") == string::npos) && (strTmp.find("metre") == string::npos))
141 {
142 // error: x-y values must be in metres
143 cerr << ERR << "GIS file x-y values (" << m_strGDALBasementDEMProjection << ") in " << m_strInitialBasementDEMFile << " must be in metres" << endl;
144 return RTN_ERR_DEMFILE;
145 }
146 }
147 }
148
149 // Now get dataset size, and do some rudimentary checks
150 m_nXGridSize = pGDALDataset->GetRasterXSize();
151
152 if (m_nXGridSize == 0)
153 {
154 // Error: silly number of columns specified
155 cerr << ERR << "invalid number of columns (" << m_nXGridSize << ") in " << m_strInitialBasementDEMFile << endl;
156 return RTN_ERR_DEMFILE;
157 }
158
159 m_nYGridSize = pGDALDataset->GetRasterYSize();
160
161 if (m_nYGridSize == 0)
162 {
163 // Error: silly number of rows specified
164 cerr << ERR << "invalid number of rows (" << m_nYGridSize << ") in " << m_strInitialBasementDEMFile << endl;
165 return RTN_ERR_DEMFILE;
166 }
167
168 // Get geotransformation info (see http://www.gdal.org/classGDALDataset.html)
169 if (CE_Failure == pGDALDataset->GetGeoTransform(m_dGeoTransform))
170 {
171 // Can't get geotransformation (note will already have sent GDAL error message to stdout)
172 cerr << ERR << CPLGetLastErrorMsg() << " in " << m_strInitialBasementDEMFile << endl;
173 return RTN_ERR_DEMFILE;
174 }
175
176 // CoastalME can only handle rasters that are oriented N-S and W-E. (If you need to work with a raster that is oriented differently, then you must rotate it before running CoastalME). So here we check whether row rotation (m_dGeoTransform[2]) and column rotation (m_dGeoTransform[4]) are both zero. See https://gdal.org/tutorials/geotransforms_tut.html
177 if ((! bFPIsEqual(m_dGeoTransform[2], 0.0, TOLERANCE)) || (! bFPIsEqual(m_dGeoTransform[4], 0.0, TOLERANCE)))
178 {
179 // Error: not oriented NS and W-E
180 cerr << ERR << m_strInitialBasementDEMFile << " is not oriented N-S and W-E. Row rotation = " << m_dGeoTransform[2] << " and column rotation = " << m_dGeoTransform[4] << endl;
181 return (RTN_ERR_RASTER_FILE_READ);
182 }
183
184 // Get the X and Y cell sizes, in external CRS units. Note that while the cell is supposed to be square, it may not be exactly so due to oddities with some GIS calculations
185 double const dCellSideX = tAbs(m_dGeoTransform[1]);
186 double const dCellSideY = tAbs(m_dGeoTransform[5]);
187
188 // Check that the cell is more or less square
189 if (! bFPIsEqual(dCellSideX, dCellSideY, 1e-2))
190 {
191 // Error: cell is not square enough
192 cerr << ERR << "cell is not square in " << m_strInitialBasementDEMFile << ", is " << dCellSideX << " x " << dCellSideY << endl;
193 return (RTN_ERR_RASTER_FILE_READ);
194 }
195
196 // Calculate the average length of cell side, the cell's diagonal, and the area of a cell (in external CRS units)
197 m_dCellSide = (dCellSideX + dCellSideY) / 2.0;
198 m_dCellArea = m_dCellSide * m_dCellSide;
199 m_dCellDiagonal = hypot(m_dCellSide, m_dCellSide);
200
201 // And calculate the inverse values
202 m_dInvCellSide = 1 / m_dCellSide;
203 m_dInvCellDiagonal = 1 / m_dCellDiagonal;
204
205 // Save some values in external CRS
206 m_dNorthWestXExtCRS = m_dGeoTransform[0] - (m_dGeoTransform[1] / 2);
207 m_dNorthWestYExtCRS = m_dGeoTransform[3] - (m_dGeoTransform[5] / 2);
208 m_dSouthEastXExtCRS = m_dGeoTransform[0] + (m_nXGridSize * m_dGeoTransform[1]) + (m_dGeoTransform[1] / 2);
209 m_dSouthEastYExtCRS = m_dGeoTransform[3] + (m_nYGridSize * m_dGeoTransform[5]) + (m_dGeoTransform[5] / 2);
210
211 // And calc the grid area in external CRS units
212 m_dExtCRSGridArea = tAbs(m_dNorthWestXExtCRS - m_dSouthEastXExtCRS) * tAbs(m_dNorthWestYExtCRS * m_dSouthEastYExtCRS);
213
214 // Now get GDAL raster band information
215 GDALRasterBand *pGDALBand = pGDALDataset->GetRasterBand(1);
216 int nBlockXSize = 0, nBlockYSize = 0;
217 pGDALBand->GetBlockSize(&nBlockXSize, &nBlockYSize);
218 m_strGDALBasementDEMDataType = GDALGetDataTypeName(pGDALBand->GetRasterDataType());
219
220 // If we have value units, then check them
221 string const strUnits = pGDALBand->GetUnitType();
222
223 if ((!strUnits.empty()) && (strUnits.find('m') == string::npos))
224 {
225 // Error: value units must be m
226 cerr << ERR << "DEM vertical units are (" << strUnits << " ) in " << m_strInitialBasementDEMFile << ", should be 'm'" << endl;
227 return RTN_ERR_DEMFILE;
228 }
229
230 // If present, get the missing value (NODATA) setting
231 CPLPushErrorHandler(CPLQuietErrorHandler); // Needed to get next line to fail silently, if it fails
232 double const dMissingValue = pGDALBand->GetNoDataValue(); // Will fail for some formats
233 CPLPopErrorHandler();
234
235 if (! bFPIsEqual(dMissingValue, m_dMissingValue, TOLERANCE))
236 {
237 cerr << " " << NOTE << "NODATA value in " << m_strInitialBasementDEMFile << " is " << dMissingValue << "\n instead using CoastalME's default floating-point NODATA value " << m_dMissingValue << endl;
238 }
239
240 // Next allocate memory for a 2D array of raster cell objects: tell the user what is happening
241 AnnounceAllocateMemory();
242 int const nRet = m_pRasterGrid->nCreateGrid();
243
244 if (nRet != RTN_OK)
245 return nRet;
246
247 // Allocate memory for a 1D floating-point array, to hold the scan line for GDAL
248 double *pdScanline = new double[m_nXGridSize];
249
250 if (NULL == pdScanline)
251 {
252 // Error, can't allocate memory
253 cerr << ERR << "cannot allocate memory for " << m_nXGridSize << " x 1D array" << endl;
254 return (RTN_ERR_MEMALLOC);
255 }
256
257 // Now read in the data
258 for (int j = 0; j < m_nYGridSize; j++)
259 {
260 // Read scanline
261 if (CE_Failure == pGDALBand->RasterIO(GF_Read, 0, j, m_nXGridSize, 1, pdScanline, m_nXGridSize, 1, GDT_Float64, 0, 0, NULL))
262 {
263 // Error while reading scanline
264 cerr << ERR << CPLGetLastErrorMsg() << " in " << m_strInitialBasementDEMFile << endl;
265 return RTN_ERR_DEMFILE;
266 }
267
268 // All OK, so read scanline into cell elevations (including any missing values)
269 for (int i = 0; i < m_nXGridSize; i++)
270 {
271 double dTmp = pdScanline[i];
272
273 if ((isnan(dTmp)) || (bFPIsEqual(dTmp, m_dGISMissingValue, TOLERANCE)))
274 dTmp = m_dMissingValue;
275
276 m_pRasterGrid->m_Cell[i][j].SetBasementElev(dTmp);
277 }
278 }
279
280 // Finished, so get rid of dataset object
281 GDALClose(pGDALDataset);
282
283 // Get rid of memory allocated to this array
284 delete[] pdScanline;
285
286 return RTN_OK;
287}

292int CSimulation::nMarkBoundingBoxEdgeCells(void)
293{
294 // The bounding box must touch the edge of the grid at least once on each side of the grid, so store these points. Search in a clockwise direction around the edge of the grid
295 vector<CGeom2DIPoint> VPtiBoundingBoxCorner;
296
297 // Start with the top (north) edge
298 bool bFound = false;
299
300 for (int nX = 0; nX < m_nXGridSize; nX++)
301 {
302 if (bFound)
303 break;
304
305 for (int nY = 0; nY < m_nYGridSize; nY++)
306 {
307 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
308 {
309 CGeom2DIPoint const PtiTmp(nX, nY);
310 VPtiBoundingBoxCorner.push_back(PtiTmp);
311 bFound = true;
312 break;
313 }
314 }
315 }
316
317 if (! bFound)
318 {
319 if (m_nLogFileDetail >= LOG_FILE_ALL)
320 LogStream << m_ulIter << ": north (top) edge of bounding box not found" << endl;
321
322 return RTN_ERR_BOUNDING_BOX;
323 }
324
325 // Do the same for the right (east) edge
326 bFound = false;
327
328 for (int nY = 0; nY < m_nYGridSize; nY++)
329 {
330 if (bFound)
331 break;
332
333 for (int nX = m_nXGridSize - 1; nX >= 0; nX--)
334 {
335 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
336 {
337 CGeom2DIPoint const PtiTmp(nX, nY);
338 VPtiBoundingBoxCorner.push_back(PtiTmp);
339 bFound = true;
340 break;
341 }
342 }
343 }
344
345 if (! bFound)
346 {
347 if (m_nLogFileDetail >= LOG_FILE_ALL)
348 LogStream << m_ulIter << ": east (right) edge of bounding box not found" << endl;
349
350 return RTN_ERR_BOUNDING_BOX;
351 }
352
353 // Do the same for the south (bottom) edge
354 bFound = false;
355
356 for (int nX = m_nXGridSize - 1; nX >= 0; nX--)
357 {
358 if (bFound)
359 break;
360
361 for (int nY = m_nYGridSize - 1; nY >= 0; nY--)
362 {
363 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
364 {
365 CGeom2DIPoint const PtiTmp(nX, nY);
366 VPtiBoundingBoxCorner.push_back(PtiTmp);
367 bFound = true;
368 break;
369 }
370 }
371 }
372
373 if (! bFound)
374 {
375 if (m_nLogFileDetail >= LOG_FILE_ALL)
376 LogStream << m_ulIter << ": south (bottom) edge of bounding box not found" << endl;
377
378 return RTN_ERR_BOUNDING_BOX;
379 }
380
381 // And finally repeat for the west (left) edge
382 bFound = false;
383
384 for (int nY = m_nYGridSize - 1; nY >= 0; nY--)
385 {
386 if (bFound)
387 break;
388
389 for (int nX = 0; nX < m_nXGridSize; nX++)
390 {
391 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
392 {
393 CGeom2DIPoint const PtiTmp(nX, nY);
394 VPtiBoundingBoxCorner.push_back(PtiTmp);
395 bFound = true;
396 break;
397 }
398 }
399 }
400
401 if (! bFound)
402 {
403 if (m_nLogFileDetail >= LOG_FILE_ALL)
404 LogStream << m_ulIter << ": west (left) edge of bounding box not found" << endl;
405
406 return RTN_ERR_BOUNDING_BOX;
407 }
408
409 // OK, so we have a point on each side of the grid, so start at this point and find the edges of the bounding box. Go round in a clockwise direction: top (north) edge first
410 for (int nX = VPtiBoundingBoxCorner[0].nGetX(); nX <= VPtiBoundingBoxCorner[1].nGetX(); nX++)
411 {
412 bFound = false;
413
414 for (int nY = VPtiBoundingBoxCorner[0].nGetY(); nY < m_nYGridSize; nY++)
415 {
416 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
417 {
418 m_ulMissingValueBasementCells++;
419 continue;
420 }
421
422 // Found a bounding box edge cell
423 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(NORTH);
424
425 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
426 m_VEdgeCellEdge.push_back(NORTH);
427
428 bFound = true;
429 break;
430 }
431
432 if (! bFound)
433 {
434 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
435 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid column " << nX << endl;
436
437 return RTN_ERR_BOUNDING_BOX;
438 }
439 }
440
441 // Right (east) edge
442 for (int nY = VPtiBoundingBoxCorner[1].nGetY(); nY <= VPtiBoundingBoxCorner[2].nGetY(); nY++)
443 {
444 bFound = false;
445
446 for (int nX = VPtiBoundingBoxCorner[1].nGetX(); nX >= 0; nX--)
447 {
448 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
449 {
450 m_ulMissingValueBasementCells++;
451 continue;
452 }
453
454 // Found a bounding box edge cell
455 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(EAST);
456
457 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
458 m_VEdgeCellEdge.push_back(EAST);
459
460 bFound = true;
461 break;
462 }
463
464 if (! bFound)
465 {
466 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
467 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid row " << nY << endl;
468
469 return RTN_ERR_BOUNDING_BOX;
470 }
471 }
472
473 // Bottom (south) edge
474 for (int nX = VPtiBoundingBoxCorner[2].nGetX(); nX >= VPtiBoundingBoxCorner[3].nGetX(); nX--)
475 {
476 bFound = false;
477
478 for (int nY = VPtiBoundingBoxCorner[2].nGetY(); nY >= 0; nY--)
479 {
480 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
481 {
482 m_ulMissingValueBasementCells++;
483 continue;
484 }
485
486 // Found a bounding box edge cell
487 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(SOUTH);
488
489 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
490 m_VEdgeCellEdge.push_back(SOUTH);
491
492 bFound = true;
493 break;
494 }
495
496 if (! bFound)
497 {
498 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
499 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid column " << nX << endl;
500
501 return RTN_ERR_BOUNDING_BOX;
502 }
503 }
504
505 // Left (west) edge
506 for (int nY = VPtiBoundingBoxCorner[3].nGetY(); nY >= VPtiBoundingBoxCorner[0].nGetY(); nY--)
507 {
508 for (int nX = VPtiBoundingBoxCorner[3].nGetX(); nX < m_nXGridSize - 1; nX++)
509 // for (int nX = VPtiBoundingBoxCorner[3].nGetX(); nX < VPtiBoundingBoxCorner[3].nGetX(); nX++)
510 {
511 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
512 {
513 m_ulMissingValueBasementCells++;
514 continue;
515 }
516
517 // Found a bounding box edge cell
518 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(WEST);
519
520 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
521 m_VEdgeCellEdge.push_back(WEST);
522
523 bFound = true;
524 break;
525 }
526
527 if (! bFound)
528 {
529 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
530 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid row " << nY << endl;
531
532 return RTN_ERR_BOUNDING_BOX;
533 }
534 }
535
536 return RTN_OK;
537}

542int CSimulation::nReadRasterGISFile(int const nDataItem, int const nLayer)
543{
544 string strGISFile;
545 string strDriverCode;
546 string strDriverDesc;
547 string strProjection;
548 string strDataType;
549
550 switch (nDataItem)
551 {
552 case (LANDFORM_RASTER):
553 // Initial Landform Class GIS data
554 strGISFile = m_strInitialLandformFile;
555 break;
556
557 case (INTERVENTION_CLASS_RASTER):
558 // Intervention class
559 strGISFile = m_strInterventionClassFile;
560 break;
561
562 case (INTERVENTION_HEIGHT_RASTER):
563 // Intervention height
564 strGISFile = m_strInterventionHeightFile;
565 break;
566
567 case (SUSP_SED_RASTER):
568 // Initial Suspended Sediment GIS data
569 strGISFile = m_strInitialSuspSedimentFile;
570 break;
571
572 case (FINE_UNCONS_RASTER):
573 // Initial Unconsolidated Fine Sediment GIS data
574 strGISFile = m_VstrInitialFineUnconsSedimentFile[nLayer];
575 break;
576
577 case (SAND_UNCONS_RASTER):
578 // Initial Unconsolidated Sand Sediment GIS data
579 strGISFile = m_VstrInitialSandUnconsSedimentFile[nLayer];
580 break;
581
582 case (COARSE_UNCONS_RASTER):
583 // Initial Unconsolidated Coarse Sediment GIS data
584 strGISFile = m_VstrInitialCoarseUnconsSedimentFile[nLayer];
585 break;
586
587 case (FINE_CONS_RASTER):
588 // Initial Consolidated Fine Sediment GIS data
589 strGISFile = m_VstrInitialFineConsSedimentFile[nLayer];
590 break;
591
592 case (SAND_CONS_RASTER):
593 // Initial Consolidated Sand Sediment GIS data
594 strGISFile = m_VstrInitialSandConsSedimentFile[nLayer];
595 break;
596
597 case (COARSE_CONS_RASTER):
598 // Initial Consolidated Coarse Sediment GIS data
599 strGISFile = m_VstrInitialCoarseConsSedimentFile[nLayer];
600 break;
601 }
602
603 // Use GDAL to create a dataset object, which then opens the GIS file
604 GDALDataset *pGDALDataset = static_cast<GDALDataset*>(GDALOpen(strGISFile.c_str(), GA_ReadOnly));
605
606 if (NULL == pGDALDataset)
607 {
608 // Can't open file (note will already have sent GDAL error message to stdout)
609 cerr << ERR << "cannot open " << strGISFile << " for input: " << CPLGetLastErrorMsg() << endl;
610 return (RTN_ERR_RASTER_FILE_READ);
611 }
612
613 // Opened OK, so get dataset information
614 strDriverCode = pGDALDataset->GetDriver()->GetDescription();
615 strDriverDesc = pGDALDataset->GetDriver()->GetMetadataItem(GDAL_DMD_LONGNAME);
616 strProjection = pGDALDataset->GetProjectionRef();
617
618 // Get geotransformation info
619 double dGeoTransform[6];
620 if (CE_Failure == pGDALDataset->GetGeoTransform(dGeoTransform))
621 {
622 // Can't get geotransformation (note will already have sent GDAL error message to stdout)
623 cerr << ERR << CPLGetLastErrorMsg() << " in " << strGISFile << endl;
624 return (RTN_ERR_RASTER_FILE_READ);
625 }
626
627 // Now get dataset size, and do some checks
628 int const nTmpXSize = pGDALDataset->GetRasterXSize();
629 if (nTmpXSize != m_nXGridSize)
630 {
631 // Error: incorrect number of columns specified
632 cerr << ERR << "different number of columns in " << strGISFile << " (" << nTmpXSize << ") and " << m_strInitialBasementDEMFile << "(" << m_nXGridSize << ")" << endl;
633 return (RTN_ERR_RASTER_FILE_READ);
634 }
635
636 int const nTmpYSize = pGDALDataset->GetRasterYSize();
637 if (nTmpYSize != m_nYGridSize)
638 {
639 // Error: incorrect number of rows specified
640 cerr << ERR << "different number of rows in " << strGISFile << " (" << nTmpYSize << ") and " << m_strInitialBasementDEMFile << " (" << m_nYGridSize << ")" << endl;
641 return (RTN_ERR_RASTER_FILE_READ);
642 }
643

875 dTmp = pdScanline[nX];
876
877 if ((isnan(dTmp)) || (bFPIsEqual(dTmp, m_dGISMissingValue, TOLERANCE)))
878 {
879 dTmp = m_dMissingValue;
880 nMissing++;
881 }
882
883 m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nLayer)->pGetUnconsolidatedSediment()->SetFineDepth(dTmp);
884 break;
885
886 case (SAND_UNCONS_RASTER):
887 // Initial Unconsolidated Sand Sediment GIS data
888 dTmp = pdScanline[nX];
889
890 if ((isnan(dTmp)) || (bFPIsEqual(dTmp, m_dGISMissingValue, TOLERANCE)))

910 break;
911

913 // Initial Consolidated Fine Sediment GIS data
914 dTmp = pdScanline[nX];
915
916 if ((isnan(dTmp)) || (bFPIsEqual(dTmp, m_dGISMissingValue, TOLERANCE)))
917 {
918 dTmp = m_dMissingValue;
919 nMissing++;
920 }

922 m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nLayer)->pGetConsolidatedSediment()->SetFineDepth(dTmp);
923 break;
924

926 // Initial Consolidated Sand Sediment GIS data
927 dTmp = pdScanline[nX];
928
929 if ((isnan(dTmp)) || (bFPIsEqual(dTmp, m_dGISMissingValue, TOLERANCE)))
930 {
931 dTmp = m_dMissingValue;
932 nMissing++;
933 }
934
935 m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nLayer)->pGetConsolidatedSediment()->SetSandDepth(dTmp);
936 break;
937
938 case (COARSE_CONS_RASTER):
939 // Initial Consolidated Coarse Sediment GIS data
940 dTmp = pdScanline[nX];

945 nMissing++;

950 }
951 }
952 }

956
957 // Get rid of memory allocated to this array
958 delete[] pdScanline;
959
960 if (nMissing > 0)
961 {
962 cerr << WARN << nMissing << " missing values in " << strGISFile << endl;
963 LogStream << WARN << nMissing << " missing values in " << strGISFile << endl;
964 }
965
966 return RTN_OK;
967}
