53void CSimulation::InitializeGDALPerformance(void)
54{
55 // Configure GDAL for optimal performance
56 // Enable GDAL threading - use all available CPU cores
57#ifdef _OPENMP
58 CPLSetConfigOption("GDAL_NUM_THREADS", "ALL_CPUS");
59#else
60 CPLSetConfigOption("GDAL_NUM_THREADS", "2"); // Fallback for non-OpenMP builds
61#endif
62
63 // Optimize GDAL memory usage and caching
64 CPLSetConfigOption("GDAL_CACHEMAX", "1024"); // 1GB cache for large grids
65 CPLSetConfigOption("GDAL_DISABLE_READDIR_ON_OPEN", "TRUE"); // Faster file access
66 CPLSetConfigOption("VSI_CACHE", "TRUE"); // Enable virtual file system cache
67 CPLSetConfigOption("VSI_CACHE_SIZE", "256000000"); // 256MB VSI cache
68
69 // Optimize grid creation performance
70 CPLSetConfigOption("GDAL_GRID_MAX_POINTS_PER_QUADTREE_LEAF", "512"); // Faster spatial indexing
71
72 // Disable GDAL warnings for cleaner output (optional)
73 // CPLSetConfigOption("CPL_LOG", "/dev/null");
74
75 LogStream << "GDAL performance optimizations enabled" << endl;
76}

81int CSimulation::nReadRasterBasementDEM(void)
82{
83 // Initialize GDAL performance settings (only needs to be done once)
84 static bool bGDALInitialized = false;
85
86 if (! bGDALInitialized)
87 {
88 InitializeGDALPerformance();
89 bGDALInitialized = true;
90 }
91
92 // Use GDAL to create a dataset object, which then opens the DEM file
93 GDALDataset * pGDALDataset = static_cast<GDALDataset*>(GDALOpen(m_strInitialBasementDEMFile.c_str(), GA_ReadOnly));
94
95 if (NULL == pGDALDataset)
96 {
97 // Can't open file (note will already have sent GDAL error message to stdout)
98 cerr << ERR << "cannot open " << m_strInitialBasementDEMFile << " for input: " << CPLGetLastErrorMsg() << endl;
99 return RTN_ERR_DEMFILE;
100 }
101
102 // Opened OK, so get GDAL basement DEM dataset information
103 m_strGDALBasementDEMDriverCode = pGDALDataset->GetDriver()->GetDescription();
104 m_strGDALBasementDEMDriverDesc = pGDALDataset->GetDriver()->GetMetadataItem(GDAL_DMD_LONGNAME);
105 m_strGDALBasementDEMProjection = pGDALDataset->GetProjectionRef();
106
107 // If we have reference units, then check that they are in metres
108 if (! m_strGDALBasementDEMProjection.empty())
109 {
110 string strTmp = strToLower(&m_strGDALBasementDEMProjection);
111
112 if ((strTmp.find("meter") == string::npos) && (strTmp.find("metre") == string::npos))
113 {
114 // error: x-y values must be in metres
115 cerr << ERR << "GIS file x-y values (" << m_strGDALBasementDEMProjection << ") in " << m_strInitialBasementDEMFile << " must be in metres" << endl;
116 return RTN_ERR_DEMFILE;
117 }
118 }
119
120 // Now get dataset size, and do some rudimentary checks
121 m_nXGridSize = pGDALDataset->GetRasterXSize();
122
123 if (m_nXGridSize == 0)
124 {
125 // Error: silly number of columns specified
126 cerr << ERR << "invalid number of columns (" << m_nXGridSize << ") in " << m_strInitialBasementDEMFile << endl;
127 return RTN_ERR_DEMFILE;
128 }
129
130 m_nYGridSize = pGDALDataset->GetRasterYSize();
131
132 if (m_nYGridSize == 0)
133 {
134 // Error: silly number of rows specified
135 cerr << ERR << "invalid number of rows (" << m_nYGridSize << ") in " << m_strInitialBasementDEMFile << endl;
136 return RTN_ERR_DEMFILE;
137 }
138
139 // Get geotransformation info (see http://www.gdal.org/classGDALDataset.html)
140 if (CE_Failure == pGDALDataset->GetGeoTransform(m_dGeoTransform))
141 {
142 // Can't get geotransformation (note will already have sent GDAL error message to stdout)
143 cerr << ERR << CPLGetLastErrorMsg() << " in " << m_strInitialBasementDEMFile << endl;
144 return RTN_ERR_DEMFILE;
145 }
146
147 // CoastalME can only handle rasters that are oriented N-S and W-E. (If you need to work with a raster that is oriented differently, then you must rotate it before running CoastalME). So here we check whether row rotation (m_dGeoTransform[2]) and column rotation (m_dGeoTransform[4]) are both zero. See https://gdal.org/tutorials/geotransforms_tut.html
148 if ((! bFPIsEqual(m_dGeoTransform[2], 0.0, TOLERANCE)) || (! bFPIsEqual(m_dGeoTransform[4], 0.0, TOLERANCE)))
149 {
150 // Error: not oriented NS and W-E
151 cerr << ERR << m_strInitialBasementDEMFile << " is not oriented N-S and W-E. Row rotation = " << m_dGeoTransform[2] << " and column rotation = " << m_dGeoTransform[4] << endl;
152 return (RTN_ERR_RASTER_FILE_READ);
153 }
154
155 // Get the X and Y cell sizes, in external CRS units. Note that while the cell is supposed to be square, it may not be exactly so due to oddities with some GIS calculations
156 double dCellSideX = tAbs(m_dGeoTransform[1]);
157 double dCellSideY = tAbs(m_dGeoTransform[5]);
158
159 // Check that the cell is more or less square
160 if (! bFPIsEqual(dCellSideX, dCellSideY, 1e-2))
161 {
162 // Error: cell is not square enough
163 cerr << ERR << "cell is not square in " << m_strInitialBasementDEMFile << ", is " << dCellSideX << " x " << dCellSideY << endl;
164 return (RTN_ERR_RASTER_FILE_READ);
165 }
166
167 // Calculate the average length of cell side, the cell's diagonal, and the area of a cell (in external CRS units)
168 m_dCellSide = (dCellSideX + dCellSideY) / 2.0;
169 m_dCellArea = m_dCellSide * m_dCellSide;
170 m_dCellDiagonal = hypot(m_dCellSide, m_dCellSide);
171
172 // And calculate the inverse values
173 m_dInvCellSide = 1 / m_dCellSide;
174 m_dInvCellDiagonal = 1 / m_dCellDiagonal;
175
176 // Save some values in external CRS
177 m_dNorthWestXExtCRS = m_dGeoTransform[0] - (m_dGeoTransform[1] / 2);
178 m_dNorthWestYExtCRS = m_dGeoTransform[3] - (m_dGeoTransform[5] / 2);
179 m_dSouthEastXExtCRS = m_dGeoTransform[0] + (m_nXGridSize * m_dGeoTransform[1]) + (m_dGeoTransform[1] / 2);
180 m_dSouthEastYExtCRS = m_dGeoTransform[3] + (m_nYGridSize * m_dGeoTransform[5]) + (m_dGeoTransform[5] / 2);
181
182 // And calc the grid area in external CRS units
183 m_dExtCRSGridArea = tAbs(m_dNorthWestXExtCRS - m_dSouthEastXExtCRS) * tAbs(m_dNorthWestYExtCRS * m_dSouthEastYExtCRS);
184
185 // Now get GDAL raster band information
186 GDALRasterBand * pGDALBand = pGDALDataset->GetRasterBand(1);
187 int nBlockXSize = 0, nBlockYSize = 0;
188 pGDALBand->GetBlockSize(&nBlockXSize, &nBlockYSize);
189 m_strGDALBasementDEMDataType = GDALGetDataTypeName(pGDALBand->GetRasterDataType());
190
191 // If we have value units, then check them
192 string strUnits = pGDALBand->GetUnitType();
193
194 if ((! strUnits.empty()) && (strUnits.find("m") == string::npos))
195 {
196 // Error: value units must be m
197 cerr << ERR << "DEM vertical units are (" << strUnits << " ) in " << m_strInitialBasementDEMFile << ", should be 'm'" << endl;
198 return RTN_ERR_DEMFILE;
199 }
200
201 // If present, get the missing value (NODATA) setting
202 CPLPushErrorHandler(CPLQuietErrorHandler); // Needed to get next line to fail silently, if it fails
203 double dMissingValue = pGDALBand->GetNoDataValue(); // Will fail for some formats
204 CPLPopErrorHandler();
205
206 if (! bFPIsEqual(dMissingValue, m_dMissingValue, TOLERANCE))
207 {
208 cerr << " " << NOTE << "NODATA value in " << m_strInitialBasementDEMFile << " is " << dMissingValue << "\n instead using CoastalME's default floating-point NODATA value " << m_dMissingValue << endl;
209 }
210
211 // Next allocate memory for a 2D array of raster cell objects: tell the user what is happening
212 AnnounceAllocateMemory();
213 int nRet = m_pRasterGrid->nCreateGrid();
214
215 if (nRet != RTN_OK)
216 return nRet;
217
218 // Allocate memory for a 1D floating-point array, to hold the scan line for GDAL
219 double* pdScanline = new double[m_nXGridSize];
220
221 if (NULL == pdScanline)
222 {
223 // Error, can't allocate memory
224 cerr << ERR << "cannot allocate memory for " << m_nXGridSize << " x 1D array" << endl;
225 return (RTN_ERR_MEMALLOC);
226 }
227
228 // Now read in the data
229 for (int j = 0; j < m_nYGridSize; j++)
230 {
231 // Read scanline
232 if (CE_Failure == pGDALBand->RasterIO(GF_Read, 0, j, m_nXGridSize, 1, pdScanline, m_nXGridSize, 1, GDT_Float64, 0, 0, NULL))
233 {
234 // Error while reading scanline
235 cerr << ERR << CPLGetLastErrorMsg() << " in " << m_strInitialBasementDEMFile << endl;
236 return RTN_ERR_DEMFILE;
237 }
238
239 // All OK, so read scanline into cell elevations (including any missing values)
240 for (int i = 0; i < m_nXGridSize; i++)
241 {
242 double dTmp = pdScanline[i];
243
244 if ((isnan(dTmp)) || (bFPIsEqual(dTmp, m_dGISMissingValue, TOLERANCE)))
245 dTmp = m_dMissingValue;
246
247 m_pRasterGrid->m_Cell[i][j].SetBasementElev(dTmp);
248 }
249 }
250
251 // Finished, so get rid of dataset object
252 GDALClose(pGDALDataset);
253
254 // Get rid of memory allocated to this array
255 delete[] pdScanline;
256
257 return RTN_OK;
258}

263int CSimulation::nMarkBoundingBoxEdgeCells(void)
264{
265 // The bounding box must touch the edge of the grid at least once on each side of the grid, so store these points. Search in a clockwise direction around the edge of the grid
266 vector<CGeom2DIPoint> VPtiBoundingBoxCorner;
267
268 // Start with the top (north) edge
269 bool bFound = false;
270
271 for (int nX = 0; nX < m_nXGridSize; nX++)
272 {
273 if (bFound)
274 break;
275
276 for (int nY = 0; nY < m_nYGridSize; nY++)
277 {
278 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
279 {
280 CGeom2DIPoint PtiTmp(nX, nY);
281 VPtiBoundingBoxCorner.push_back(PtiTmp);
282 bFound = true;
283 break;
284 }
285 }
286 }
287
288 if (! bFound)
289 {
290 if (m_nLogFileDetail >= LOG_FILE_ALL)
291 LogStream << m_ulIter << ": north (top) edge of bounding box not found" << endl;
292
293 return RTN_ERR_BOUNDING_BOX;
294 }
295
296 // Do the same for the right (east) edge
297 bFound = false;
298
299 for (int nY = 0; nY < m_nYGridSize; nY++)
300 {
301 if (bFound)
302 break;
303
304 for (int nX = m_nXGridSize - 1; nX >= 0; nX--)
305 {
306 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
307 {
308 CGeom2DIPoint PtiTmp(nX, nY);
309 VPtiBoundingBoxCorner.push_back(PtiTmp);
310 bFound = true;
311 break;
312 }
313 }
314 }
315
316 if (! bFound)
317 {
318 if (m_nLogFileDetail >= LOG_FILE_ALL)
319 LogStream << m_ulIter << ": east (right) edge of bounding box not found" << endl;
320
321 return RTN_ERR_BOUNDING_BOX;
322 }
323
324 // Do the same for the south (bottom) edge
325 bFound = false;
326
327 for (int nX = m_nXGridSize - 1; nX >= 0; nX--)
328 {
329 if (bFound)
330 break;
331
332 for (int nY = m_nYGridSize - 1; nY >= 0; nY--)
333 {
334 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
335 {
336 CGeom2DIPoint PtiTmp(nX, nY);
337 VPtiBoundingBoxCorner.push_back(PtiTmp);
338 bFound = true;
339 break;
340 }
341 }
342 }
343
344 if (! bFound)
345 {
346 if (m_nLogFileDetail >= LOG_FILE_ALL)
347 LogStream << m_ulIter << ": south (bottom) edge of bounding box not found" << endl;
348
349 return RTN_ERR_BOUNDING_BOX;
350 }
351
352 // And finally repeat for the west (left) edge
353 bFound = false;
354
355 for (int nY = m_nYGridSize - 1; nY >= 0; nY--)
356 {
357 if (bFound)
358 break;
359
360 for (int nX = 0; nX < m_nXGridSize; nX++)
361 {
362 if (! m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
363 {
364 CGeom2DIPoint PtiTmp(nX, nY);
365 VPtiBoundingBoxCorner.push_back(PtiTmp);
366 bFound = true;
367 break;
368 }
369 }
370 }
371
372 if (! bFound)
373 {
374 if (m_nLogFileDetail >= LOG_FILE_ALL)
375 LogStream << m_ulIter << ": west (left) edge of bounding box not found" << endl;
376
377 return RTN_ERR_BOUNDING_BOX;
378 }
379
380 // OK, so we have a point on each side of the grid, so start at this point and find the edges of the bounding box. Go round in a clockwise direction: top (north) edge first
381 for (int nX = VPtiBoundingBoxCorner[0].nGetX(); nX <= VPtiBoundingBoxCorner[1].nGetX(); nX++)
382 {
383 bFound = false;
384
385 for (int nY = VPtiBoundingBoxCorner[0].nGetY(); nY < m_nYGridSize; nY++)
386 {
387 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
388 {
389 m_ulMissingValueBasementCells++;
390 continue;
391 }
392
393 // Found a bounding box edge cell
394 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(NORTH);
395
396 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
397 m_VEdgeCellEdge.push_back(NORTH);
398
399 bFound = true;
400 break;
401 }
402
403 if (! bFound)
404 {
405 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
406 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid column " << nX << endl;
407
408 return RTN_ERR_BOUNDING_BOX;
409 }
410 }
411
412 // Right (east) edge
413 for (int nY = VPtiBoundingBoxCorner[1].nGetY(); nY <= VPtiBoundingBoxCorner[2].nGetY(); nY++)
414 {
415 bFound = false;
416
417 for (int nX = VPtiBoundingBoxCorner[1].nGetX(); nX >= 0; nX--)
418 {
419 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
420 {
421 m_ulMissingValueBasementCells++;
422 continue;
423 }
424
425 // Found a bounding box edge cell
426 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(EAST);
427
428 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
429 m_VEdgeCellEdge.push_back(EAST);
430
431 bFound = true;
432 break;
433 }
434
435 if (! bFound)
436 {
437 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
438 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid row " << nY << endl;
439
440 return RTN_ERR_BOUNDING_BOX;
441 }
442 }
443
444 // Bottom (south) edge
445 for (int nX = VPtiBoundingBoxCorner[2].nGetX(); nX >= VPtiBoundingBoxCorner[3].nGetX(); nX--)
446 {
447 bFound = false;
448
449 for (int nY = VPtiBoundingBoxCorner[2].nGetY(); nY >= 0; nY--)
450 {
451 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
452 {
453 m_ulMissingValueBasementCells++;
454 continue;
455 }
456
457 // Found a bounding box edge cell
458 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(SOUTH);
459
460 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
461 m_VEdgeCellEdge.push_back(SOUTH);
462
463 bFound = true;
464 break;
465 }
466
467 if (! bFound)
468 {
469 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
470 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid column " << nX << endl;
471
472 return RTN_ERR_BOUNDING_BOX;
473 }
474 }
475
476 // Left (west) edge
477 for (int nY = VPtiBoundingBoxCorner[3].nGetY(); nY >= VPtiBoundingBoxCorner[0].nGetY(); nY--)
478 {
479 for (int nX = VPtiBoundingBoxCorner[3].nGetX(); nX < m_nXGridSize - 1; nX++)
480 // for (int nX = VPtiBoundingBoxCorner[3].nGetX(); nX < VPtiBoundingBoxCorner[3].nGetX(); nX++)
481 {
482 if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
483 {
484 m_ulMissingValueBasementCells++;
485 continue;
486 }
487
488 // Found a bounding box edge cell
489 m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(WEST);
490
491 m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
492 m_VEdgeCellEdge.push_back(WEST);
493
494 bFound = true;
495 break;
496 }
497
498 if (! bFound)
499 {
500 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
501 LogStream << m_ulIter << ": could not find a bounding box edge cell for grid row " << nY << endl;
502
503 return RTN_ERR_BOUNDING_BOX;
504 }
505 }
506
507 return RTN_OK;
508}

513int CSimulation::nReadRasterGISFile(int const nDataItem, int const nLayer)
514{
515 string
516 strGISFile,
517 strDriverCode,
518 strDriverDesc,
519 strProjection,
520 strDataType;
521
522 switch (nDataItem)
523 {
524 case (LANDFORM_RASTER):
525 // Initial Landform Class GIS data
526 strGISFile = m_strInitialLandformFile;
527 break;
528
529 case (INTERVENTION_CLASS_RASTER):
530 // Intervention class
531 strGISFile = m_strInterventionClassFile;
532 break;
533
534 case (INTERVENTION_HEIGHT_RASTER):
535 // Intervention height
536 strGISFile = m_strInterventionHeightFile;
537 break;
538
539 case (SUSP_SED_RASTER):
540 // Initial Suspended Sediment GIS data
541 strGISFile = m_strInitialSuspSedimentFile;
542 break;
543
544 case (FINE_UNCONS_RASTER):
545 // Initial Unconsolidated Fine Sediment GIS data
546 strGISFile = m_VstrInitialFineUnconsSedimentFile[nLayer];
547 break;
548
549 case (SAND_UNCONS_RASTER):
550 // Initial Unconsolidated Sand Sediment GIS data
551 strGISFile = m_VstrInitialSandUnconsSedimentFile[nLayer];
552 break;
553
554 case (COARSE_UNCONS_RASTER):
555 // Initial Unconsolidated Coarse Sediment GIS data
556 strGISFile = m_VstrInitialCoarseUnconsSedimentFile[nLayer];
557 break;
558
559 case (FINE_CONS_RASTER):
560 // Initial Consolidated Fine Sediment GIS data
561 strGISFile = m_VstrInitialFineConsSedimentFile[nLayer];
562 break;
563
564 case (SAND_CONS_RASTER):
565 // Initial Consolidated Sand Sediment GIS data
566 strGISFile = m_VstrInitialSandConsSedimentFile[nLayer];
567 break;
568
569 case (COARSE_CONS_RASTER):
570 // Initial Consolidated Coarse Sediment GIS data
571 strGISFile = m_VstrInitialCoarseConsSedimentFile[nLayer];
572 break;
573 }
574
575 // Do we have a filename for this data item? If we don't then just return
576 if (! strGISFile.empty())
577 {
578 // We do have a filename, so use GDAL to create a dataset object, which then opens the GIS file
579 GDALDataset * pGDALDataset = static_cast<GDALDataset*>(GDALOpen(strGISFile.c_str(), GA_ReadOnly));
580
581 if (NULL == pGDALDataset)
582 {
583 // Can't open file (note will already have sent GDAL error message to stdout)
584 cerr << ERR << "cannot open " << strGISFile << " for input: " << CPLGetLastErrorMsg() << endl;
585 return (RTN_ERR_RASTER_FILE_READ);
586 }
587
588 // Opened OK, so get dataset information
589 strDriverCode = pGDALDataset->GetDriver()->GetDescription();
590 strDriverDesc = pGDALDataset->GetDriver()->GetMetadataItem(GDAL_DMD_LONGNAME);
591 strProjection = pGDALDataset->GetProjectionRef();
592
593 // Get geotransformation info
594 double dGeoTransform[6];
595
596 if (CE_Failure == pGDALDataset->GetGeoTransform(dGeoTransform))
597 {
598 // Can't get geotransformation (note will already have sent GDAL error message to stdout)
599 cerr << ERR << CPLGetLastErrorMsg() << " in " << strGISFile << endl;
600 return (RTN_ERR_RASTER_FILE_READ);
601 }
602
603 // Now get dataset size, and do some checks
604 int nTmpXSize = pGDALDataset->GetRasterXSize();
605
606 if (nTmpXSize != m_nXGridSize)
607 {
608 // Error: incorrect number of columns specified
609 cerr << ERR << "different number of columns in " << strGISFile << " (" << nTmpXSize << ") and " << m_strInitialBasementDEMFile << "(" << m_nXGridSize << ")" << endl;
610 return (RTN_ERR_RASTER_FILE_READ);
611 }
612
613 int nTmpYSize = pGDALDataset->GetRasterYSize();
614
615 if (nTmpYSize != m_nYGridSize)
616 {
617 // Error: incorrect number of rows specified
618 cerr << ERR << "different number of rows in " << strGISFile << " (" << nTmpYSize << ") and " << m_strInitialBasementDEMFile << " (" << m_nYGridSize << ")" << endl;
619 return (RTN_ERR_RASTER_FILE_READ);
620 }
621
622 double dTmp = m_dGeoTransform[0] - (m_dGeoTransform[1] / 2);
623
624 if (! bFPIsEqual(dTmp, m_dNorthWestXExtCRS, TOLERANCE))
625 {
626 // Error: different min x from DEM file
627 cerr << ERR << "different min x values in " << strGISFile << " (" << dTmp << ") and " << m_strInitialBasementDEMFile << " (" << m_dNorthWestXExtCRS << ")" << endl;
628 return (RTN_ERR_RASTER_FILE_READ);
629 }
630
631 dTmp = m_dGeoTransform[3] - (m_dGeoTransform[5] / 2);
632
633 if (! bFPIsEqual(dTmp, m_dNorthWestYExtCRS, TOLERANCE))
634 {
635 // Error: different min x from DEM file
636 cerr << ERR << "different min y values in " << strGISFile << " (" << dTmp << ") and " << m_strInitialBasementDEMFile << " (" << m_dNorthWestYExtCRS << ")" << endl;
