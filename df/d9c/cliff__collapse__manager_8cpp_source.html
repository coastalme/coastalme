63int CCliffCollapseManager::ProcessAllCliffCollapse()
64{
65 if (! m_bCliffCollapseEnabled || !m_pCliffAlgorithm)
66 return RTN_OK;
67
68 // Process cliff collapse for all coasts
69 int nCoasts = static_cast<int>(m_pSimulation->m_VCoast.size());
70
71 // Use OpenMP to parallelize coast processing
72#ifdef _OPENMP
73#pragma omp parallel for
74#endif
75 for (int nCoast = 0; nCoast < nCoasts; nCoast++)
76 {
77 ProcessCoastCliffCollapse(nCoast);
78 }
79
80 return RTN_OK;

84int CCliffCollapseManager::ProcessCoastCliffCollapse(int nCoast)
85{
86 if (! m_bCliffCollapseEnabled || !m_pCliffAlgorithm)
87 return RTN_OK;
88
89 // Get the coast object
90 if (nCoast >= static_cast<int>(m_pSimulation->m_VCoast.size()))
91 return RTN_ERR_NOCOAST;
92
93 CRWCoast* pCoast = &(m_pSimulation->m_VCoast[nCoast]);
94
95 // Process cliff collapse for all points on this coast
96 int nCoastPoints = pCoast->nGetCoastlineSize();
97
98 // Use OpenMP to parallelize point processing within each coast
99#ifdef _OPENMP
100#pragma omp parallel for
101#endif
102 for (int nPoint = 0; nPoint < nCoastPoints; nPoint++)
103 {
104 ProcessPointCliffCollapse(nCoast, nPoint);
105 }
106
107 return RTN_OK;
108}

111int CCliffCollapseManager::ProcessPointCliffCollapse(int nCoast, int nPointOnCoast)
112{
113 if (! m_bCliffCollapseEnabled || !m_pCliffAlgorithm)
114 return RTN_OK;
115
116 // Extract cliff data from simulation
117 CCliffData cliffData = ExtractCliffData(nCoast, nPointOnCoast);
118
119 // Skip if no cliff data available
120 if (bFPIsEqual(cliffData.CliffToe.dGetX(), 0.0, TOLERANCE) && bFPIsEqual(cliffData.CliffToe.dGetY(), 0.0, TOLERANCE))
121 return RTN_OK;
122
123 // Process the cliff collapse algorithm
124 CCliffResults results = m_pCliffAlgorithm->ProcessTimestep(cliffData);
125
126 // Apply results back to simulation
127 ApplyCliffResults(results, nCoast, nPointOnCoast);
128
129 return RTN_OK;
130}

175bool CCliffCollapseManager::SwitchAlgorithm(const string& strNewAlgorithmName)
176{
177 // Create new algorithm
178 auto pNewAlgorithm = CCliffAlgorithmFactory::CreateAlgorithm(strNewAlgorithmName);
179 if (! pNewAlgorithm)
180 return false;
181
182 // Initialize new algorithm
183 if (! pNewAlgorithm->Initialize(strNewAlgorithmName))
184 return false;
185
186 // Replace old algorithm
187 m_pCliffAlgorithm = std::move(pNewAlgorithm);
188 m_strAlgorithmName = strNewAlgorithmName;
189
190 return true;
191}
