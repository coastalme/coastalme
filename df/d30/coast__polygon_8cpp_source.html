770CGeom2DIPoint CGeomCoastPolygon::PtiGetFillStartPoint(void)
771{
772 int nVertices = static_cast<int>(m_VPtiVertices.size());
773 double dXTot = 0;
774 double dYTot = 0;
775
776 for (int n = 0; n < nVertices; n++)
777 {
778 dXTot += m_VPtiVertices[n].nGetX();
779 dYTot += m_VPtiVertices[n].nGetY();
780 }
781 return CGeom2DIPoint(nRound(dXTot / nVertices), nRound(dYTot / nVertices));
