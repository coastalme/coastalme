795CGeom2DIPoint CGeomCoastPolygon::PtiGetFillStartPoint(void)
796{
797 int const nVertices = static_cast<int>(m_VPtiVertices.size());
798 double dXTot = 0;
799 double dYTot = 0;
800
801 for (int n = 0; n < nVertices; n++)
802 {
803 dXTot += m_VPtiVertices[n].nGetX();
804 dYTot += m_VPtiVertices[n].nGetY();
805 }
806
807 return CGeom2DIPoint(nRound(dXTot / nVertices), nRound(dYTot / nVertices));
