80bool bIsStringValidInt(string& str)
81{
82 // Trim leading whitespace
83 size_t nPos = str.find_first_not_of(" \t");
84 if (nPos != string::npos)
85 str = str.substr(nPos);
86
87 // If the first character is the sign, remove it
88 if ((str[0] == '-') || (str[0] == '+'))
89 str.erase(0, 1);
90
91 // Now check that the string contains only numbers
92 return (str.find_first_not_of("0123456789") == string::npos);
93}

208string strRightPerCent(double const d1, double const d2, int const nWidth, int const nDigits, bool const bShowDash)
209{
210 stringstream ss;
211 ss << std::fixed << std::right;
212
213 // Are either of the inputs zero?
214 if ((bFPIsEqual(d1, 0.0, TOLERANCE)) || (bFPIsEqual(d2, 0.0, TOLERANCE)))
215 {
216 ss.fill(' ');
217 ss.width(nWidth-1);
218
219 if (bShowDash)
220 ss << "-";
221 else
222 ss << SPACE;
223 }
224 else
225 {
226 // Non-zero, so calculate the percentage
227 double dResult = 100 * d1 / d2;
228
229 stringstream ssResult;
230 ssResult << std::fixed << std::right;
231 ssResult.precision(nDigits);
232 ssResult << "(" << dResult << "%)";
233
234 long int nResultWidth = ssResult.str().size();
235
236 for (int i = 0; i < (nWidth - nResultWidth - 1); i++)
237 ss << SPACE;
238
239 ss << ssResult.str();
240 }
241
242 // Append a final space
243 ss << " ";
244
245 return ss.str();
246}
