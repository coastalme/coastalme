613int CSimulation::nDoSimulation(int nArg, char const* pcArgv[])
614{
615 // ================================================== initialisation section ================================================
616 // Hello, World!
617 AnnounceStart();
618
619 // Start the clock ticking
620 StartClock();
621
622 // Deal with command-line parameters
623 int nRet = nHandleCommandLineParams(nArg, pcArgv);
624
625 if (nRet != RTN_OK)
626 return (nRet);
627
628 // Find out the folder in which the CoastalME executable sits, in order to open the .ini file (they are assumed to be in the same folder)
629 if (! bFindExeDir(pcArgv[0]))
630 return (RTN_ERR_CMEDIR);
631
632 // OK, we are off, tell the user about the licence and the start time
633 AnnounceLicence();
634
635 // Read the .ini file and get the name of the run-data file, and path for output etc.
636 if (! bReadIniFile())
637 return (RTN_ERR_INI);
638
639 // Check if output dir exists
640 if ((! is_directory(m_strOutPath.c_str())) || (! exists(m_strOutPath.c_str())))
641 {
642 // Output dir does not exist
643 bool bCreateDir = false;
644
645 if ((isatty(fileno(stdout))) && (isatty(fileno(stderr))))
646 {
647 // Running with stdout and stderr as a tty, so ask the user if they wish to create it
648 char ch;
649 cerr << endl
650 << "Output folder '" << m_strOutPath << "' does not exist. Create it? (Y/N) ";
651 cerr.flush();
652 cin.get(ch);
653
654 if ((ch == 'y') || (ch == 'Y'))
655 bCreateDir = true;
656 }
657
658 else
659 {
660 // Running with stdout or stderr not a tty, so create output dir rather than abort
661 bCreateDir = true;
662 }
663
664 if (bCreateDir)
665 {
666 // Yes, so create the directory
667 create_directories(m_strOutPath.c_str());
668 cerr << m_strOutPath << " created" << endl << endl;
669 }
670
671 else
672 // Nope, just end the run
673 return RTN_USER_ABORT;
674 }
675
676 // We have the name of the run-data input file, so read it
677 if (! bReadRunDataFile())
678 return RTN_ERR_RUNDATA;
679
680 // Check raster GIS output format
681 if (! bCheckRasterGISOutputFormat())
682 return (RTN_ERR_RASTER_GIS_OUT_FORMAT);
683
684 // Check vector GIS output format
685 if (! bCheckVectorGISOutputFormat())
686 return (RTN_ERR_VECTOR_GIS_OUT_FORMAT);
687
688 // Open log file
689 if (! bOpenLogFile())
690 return (RTN_ERR_LOGFILE);
691
692 // Set up the time series output files
693 if (! bSetUpTSFiles())
694 return (RTN_ERR_TSFILE);
695
696 // Initialize the random number generators
697 for (int n = 0; n < NUMBER_OF_RNGS; n++)
698 m_Rand[n].seed(m_ulRandSeed[n]);
699
700 // If we are doing Savitzky-Golay smoothing of the vector coastline(s), calculate the filter coefficients
701 if (m_nCoastSmooth == SMOOTH_SAVITZKY_GOLAY)
702 CalcSavitzkyGolayCoeffs();
703
704 // Create the raster grid object
705 m_pRasterGrid = new CGeomRasterGrid(this);
706
707 // Create the cliff collapse manager object
708 m_pCliffCollapseManager = new CCliffCollapseManager(this);
709
710 // Read in the basement layer (must have this file), create the raster grid, then read in the basement DEM data to the array
711 AnnounceReadBasementDEM();
712 nRet = nReadRasterBasementDEM();
713
714 if (nRet != RTN_OK)
715 return nRet;
716
717 // If we are simulating cliff collapse: then now that we have a value for m_dCellSide, we can check some more input parameters. Talus must be more than one cell wide, and since the number of cells must be odd, three cells is the minimum width
