609int CSimulation::nDoSimulation(int nArg, char const* pcArgv[])
610{
611 // ================================================== initialization section ================================================
612 // Hello, World!
613 AnnounceStart();
614
615 // Start the clock ticking
616 StartClock();
617
618 // Deal with command-line parameters
619 int nRet = nHandleCommandLineParams(nArg, pcArgv);
620
621 if (nRet != RTN_OK)
622 return (nRet);
623
624 // Find out the folder in which the CoastalME executable sits, in order to open the .ini file (they are assumed to be in the same folder)
625 if (!bFindExeDir(pcArgv[0]))
626 return (RTN_ERR_CMEDIR);
627
628 // OK, we are off, tell the user about the licence and the start time
629 AnnounceLicence();
630
631 // Read the .ini file and get the name of the run-data file, and path for output etc.
632 if (!bReadIniFile())
633 return (RTN_ERR_INI);
634
635 // Check if output dir exists
636 if ((!is_directory(m_strOutPath.c_str())) || (!exists(m_strOutPath.c_str())))
637 {
638 // Output dir does not exist
639 bool bCreateDir = false;
640
641 if ((isatty(fileno(stdout))) && (isatty(fileno(stderr))))
642 {
643 // Running with stdout and stderr as a tty, so ask the user if they wish to create it
644 char ch;
645 cerr << endl
646 << "Output folder '" << m_strOutPath << "' does not exist. Create it? (Y/N) ";
647 cerr.flush();
648 cin.get(ch);
649
650 if ((ch == 'y') || (ch == 'Y'))
651 bCreateDir = true;
652 }
653
654 else
655 {
656 // Running with stdout or stderr not a tty, so create output dir rather than abort
657 bCreateDir = true;
658 }
659
660 if (bCreateDir)
661 {
662 // Yes, so create the directory
663 create_directories(m_strOutPath.c_str());
664 cerr << m_strOutPath << " created" << endl
665 << endl;
666 }
667
668 else
669 // Nope, just end the run
670 return RTN_USER_ABORT;
671 }
672
673 // We have the name of the run-data input file, so read it
674 if (!bReadRunDataFile())
675 return RTN_ERR_RUNDATA;
676
677 // Check raster GIS output format
678 if (!bCheckRasterGISOutputFormat())
679 return (RTN_ERR_RASTER_GIS_OUT_FORMAT);
680
681 // Check vector GIS output format
682 if (!bCheckVectorGISOutputFormat())
683 return (RTN_ERR_VECTOR_GIS_OUT_FORMAT);
684
685 // Open log file
686 if (!bOpenLogFile())
687 return (RTN_ERR_LOGFILE);
688
689 // Set up the time series output files
690 if (!bSetUpTSFiles())
691 return (RTN_ERR_TSFILE);
692
693 // Initialize the random number generators
694 for (int n = 0; n < NUMBER_OF_RNGS; n++)
695 m_Rand[n].seed(m_ulRandSeed[n]);
696
697 // If we are doing Savitzky-Golay smoothing of the vector coastline(s), calculate the filter coefficients
698 if (m_nCoastSmooth == SMOOTH_SAVITZKY_GOLAY)
699 CalcSavitzkyGolayCoeffs();
700
701 // Create the raster grid object
702 m_pRasterGrid = new CGeomRasterGrid(this);
703
704 // Read in the basement layer (must have this file), create the raster grid, then read in the basement DEM data to the array
705 AnnounceReadBasementDEM();
706 nRet = nReadRasterBasementDEM();
707
708 if (nRet != RTN_OK)
709 return nRet;
710
711 // If we are simulating cliff collapse: then now that we have a value for m_dCellSide, we can check some more input parameters. Talus must be more than one cell wide, and since the number of cells must be odd, three cells is the minimum width
712 if (m_bDoCliffCollapse)
713 {
714 int const nTmp = nConvertMetresToNumCells(m_dCliffDepositionPlanviewWidth);
715
