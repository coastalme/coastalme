231bool CGeomProfile::bProfileOK(void) const
232{
233 // All profiles without problems, but not start- or end-of-coast profiles
234 if ((! m_bStartOfCoast) &&
235 (! m_bEndOfCoast) &&
236 (! m_bHitLand) &&
237 (! m_bHitIntervention) &&
238 (! m_bHitCoast) &&
239 (! m_bTooShort) &&
240 (! m_bTruncatedSameCoast) &&
241 (! m_bHitAnotherProfile) &&
242 (! m_bCShoreProblem))
243 return true;
244
245 return false;

323bool CGeomProfile::bInsertIntersection(double const dX, double const dY, int const nSeg)
324{
325 // Safety check
326 if (nSeg >= nGetNumLineSegments())
327 return false;
328
329 vector<CGeom2DPoint>::iterator it;
330 it = CGeomMultiLine::m_VPoints.begin();
331
332 // Do the insertion
333 CGeomMultiLine::m_VPoints.insert(it + nSeg + 1, CGeom2DPoint(dX, dY));
334
335 // Now insert a line segment in the associated multi-line, this will inherit the profile/line seg details from the preceding line segment
336 CGeomMultiLine::InsertLineSegment(nSeg);
337
338 return true;
339}

399bool CGeomProfile::bIsPointInProfile(double const dX, double const dY, int& nPoint)
400{
401 CGeom2DPoint const Pt(dX, dY);
402 auto it = find(CGeomMultiLine::m_VPoints.begin(), CGeomMultiLine::m_VPoints.end(), &Pt);

542int CGeomProfile::nGetCellGivenDepth(CGeomRasterGrid const* pGrid, double const dDepthIn)
543{
544 int nIndex = INT_NODATA; // If not found, i.e. if every profile cell has sea depth less than dDepthIn
545
546 for (unsigned int n = 0; n < m_VCellInProfile.size(); n++)
547 {
548 int const nX = m_VCellInProfile[n].nGetX();
549 int const nY = m_VCellInProfile[n].nGetY();
550
551 double const dCellDepth = pGrid->m_Cell[nX][nY].dGetSeaDepth();
552
553 if (dCellDepth >= dDepthIn)
554 {
555 nIndex = n;
556
557 if (n > 0)
558 nIndex = n - 1; // Grid CRS units
559
560 break;
561 }
562 }
563
564 return nIndex;
565}
