148bool CGeomProfile::bProfileOK(void) const
149{
150 // All profiles without problems, but not start- or end-of-coast profiles
151 if ((! m_bStartOfCoast) &&
152 (! m_bEndOfCoast) &&
153 (! m_bHitLand) &&
154 (! m_bHitCoast) &&
155 (! m_bTooShort) &&
156 (! m_bTruncated) &&
157 (! m_bHitAnotherProfile))
158 return true;
159
160 return false;
161}

218bool CGeomProfile::bInsertIntersection(double const dX, double const dY, int const nSeg)
219{
220 // Safety check
221 if (nSeg >= nGetNumLineSegments())
222 return false;
223
224 vector<CGeom2DPoint>::iterator it;
225 it = m_VPoints.begin();
226
227 // Do the insertion
228 m_VPoints.insert(it + nSeg + 1, CGeom2DPoint(dX, dY));
229
230 // Now insert a line segment in the associated multi-line, this will inherit the profile/line seg details from the preceding line segment
231 CGeomMultiLine::InsertLineSegment(nSeg);
232
233 return true;
234}

293bool CGeomProfile::bIsPointInProfile(double const dX, double const dY, int &nPoint)
294{
295 CGeom2DPoint Pt(dX, dY);
296 auto it = find(m_VPoints.begin(), m_VPoints.end(), &Pt);
297 if (it != m_VPoints.end())
298 {
299 // Found, so return true and set nPoint to be the index of the point which was found
300 nPoint = static_cast<int>(it - m_VPoints.begin());
301 return true;
302 }
303 else

451int CGeomProfile::nGetCellGivenDepth(CGeomRasterGrid const* pGrid, double const dDepthIn)
452{
453 int nIndex = INT_NODATA; // If not found, i.e. if every profile cell has sea depth less than dDepthIn
454
455 for (unsigned int n = 0; n < m_VCellInProfile.size(); n++)
456 {
457 int nX = m_VCellInProfile[n].nGetX();
458 int nY = m_VCellInProfile[n].nGetY();
459
460 double dCellDepth = pGrid->m_Cell[nX][nY].dGetSeaDepth();
461 if (dCellDepth >= dDepthIn)
462 {
463 nIndex = n;
464
465 if (n > 0)
466 nIndex = n - 1; // Grid CRS units
467
468 break;
469 }
470 }
471
472 if (nIndex == INT_NODATA)
473 {
474 nIndex = static_cast<int>(m_VCellInProfile.size()) - 1;
475 }
476
477 return nIndex;
478}
