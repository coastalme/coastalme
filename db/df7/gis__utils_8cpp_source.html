206//===============================================================================================================================
207void CSimulation::KeepWithinValidGrid(int nX0, int nY0, int &nX1, int &nY1) const
208{
209 // Safety check: make sure that the first point is within the valid grid
210 if (nX0 >= m_nXGridSize)
211 nX0 = m_nXGridSize - 1;
212 else if (nX0 < 0)
213 nX0 = 0;
214
215 if (nY0 >= m_nYGridSize)
216 nY0 = m_nYGridSize - 1;
217 else if (nY0 < 0)
218 nY0 = 0;
219
220 // OK let's go
221 int nDiffX = nX0 - nX1;
222 int nDiffY = nY0 - nY1;
223
224 if (nDiffX == 0)
225 {
226 // The two points have the same x coordinates, so we just need to constrain the y co-ord
227 if (nY1 < nY0)
228 {
229 nY1 = -1;
230
231 do
232 {
233 nY1++;
234 } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
235
236 return;
237 }
238 else
239 {
240 nY1 = m_nYGridSize;
241
242 do
243 {
244 nY1--;
245 } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
246
247 return;
248 }
249 }
250 else if (nDiffY == 0)
251 {
252 // The two points have the same y coordinates, so we just need to constrain the x co-ord
253 if (nX1 < nX0)
254 {
255 nX1 = -1;
256
257 do
258 {
259 nX1++;
260 } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
261
262 return;
263 }
264 else
265 {
266 nX1 = m_nXGridSize;
267
268 do
269 {
270 nX1--;
271 } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
272
273 return;
274 }
275 }
276 else
277 {
278 // The two points have different x coordinates and different y coordinates, so we have to work harder. First find which of the coordinates is the greatest distance outside the grid, and constrain that co-ord for efficiency (since this will reduce the number of times round the loop). Note that both may be inside the grid, if the incorrect co-ord is in the invalid margin, in which case arbitrarily contrain the x co-ord
279 int nXDistanceOutside = 0;
280 int nYDistanceOutside = 0;
281
282 if (nX1 < 0)
283 nXDistanceOutside = -nX1;
284 else if (nX1 >= m_nXGridSize)
285 nXDistanceOutside = nX1 - m_nXGridSize + 1;
286
287 if (nY1 < 0)
288 nYDistanceOutside = -nY1;
289 else if (nY1 >= m_nYGridSize)
290 nXDistanceOutside = nY1 - m_nYGridSize + 1;
291
292 if (nXDistanceOutside >= nYDistanceOutside)
293 {
294 // Constrain the x co-ord
295 if (nX1 < nX0)
296 {
297 // The incorrect x co-ord is less than the correct x co-ord: constrain it and find the y co-ord
298 nX1 = -1;
299
300 do
301 {
302 nX1++;
303
304 nY1 = nY0 + nRound(((nX1 - nX0) * nDiffY) / static_cast<double>(nDiffX));
305 } while ((nY1 < 0) || (nY1 >= m_nYGridSize) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
306
307 return;
308 }
309
310 else
311 {
312 // The incorrect x co-ord is greater than the correct x-co-ord: constrain it and find the y co-ord
313 nX1 = m_nXGridSize;
314
315 do
316 {
317 nX1--;
318
319 nY1 = nY0 + nRound(((nX1 - nX0) * nDiffY) / static_cast<double>(nDiffX));
320 } while ((nY1 < 0) || (nY1 >= m_nYGridSize) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
321
322 return;
323 }
324 }
325 else
326 {
327 // Constrain the y co-ord
328 if (nY1 < nY0)
329 {
330 // The incorrect y co-ord is less than the correct y-co-ord: constrain it and find the x co-ord
331 nY1 = -1;
332
333 do
334 {
335 nY1++;
336
337 nX1 = nX0 + nRound(((nY1 - nY0) * nDiffX) / static_cast<double>(nDiffY));
338 } while ((nX1 < 0) || (nX1 >= m_nXGridSize) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
339
340 return;
341 }
342
343 else
344 {
345 // The incorrect y co-ord is greater than the correct y co-ord: constrain it and find the x co-ord
346 nY1 = m_nYGridSize;
347
348 do
349 {
350 nY1--;
351
352 nX1 = nX0 + nRound(((nY1 - nY0) * nDiffX) / static_cast<double>(nDiffY));
353 } while ((nX1 < 0) || (nX1 >= m_nXGridSize) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
354
355 return;
356 }
357 }
358 }

364double CSimulation::dKeepWithin360(double const dAngle)
365{
366 double dNewAngle = dAngle;
367
368 // Sort out -ve angles
369 while (dNewAngle < 0)
370 dNewAngle += 360;
371
372 // Sort out angles > 360
373 while (dNewAngle > 360)
374 dNewAngle -= 360;
375
376 return dNewAngle;

474//===============================================================================================================================
476//===============================================================================================================================
477CGeom2DIPoint CSimulation::PtiPolygonCentroid(vector<CGeom2DIPoint>* pVIn)
478{
479 CGeom2DIPoint PtiCentroid(0, 0);
480 int nSize = static_cast<int>(pVIn->size());
481 int nX0 = 0; // Current vertex X
482 int nY0 = 0; // Current vertex Y
483 int nX1 = 0; // Next vertex X
484 int nY1 = 0; // Next vertex Y
485
486 double dA = 0; // Partial signed area
487 double dSignedArea = 0.0;
488
489 // For all vertices except last
490 for (int i = 0; i < nSize - 1; ++i)
491 {
492 nX0 = pVIn->at(i).nGetX();
493 nY0 = pVIn->at(i).nGetY();
494 nX1 = pVIn->at(i + 1).nGetX();
495 nY1 = pVIn->at(i + 1).nGetY();
496
497 dA = (nX0 * nY1) - (nX1 * nY0);
498 dSignedArea += dA;
499 PtiCentroid.AddXAddY((nX0 + nX1) * dA, (nY0 + nY1) * dA);
500 }
501
502 // Do last vertex separately to avoid performing an expensive modulus operation in each iteration
503 nX0 = pVIn->at(nSize - 1).nGetX();
504 nY0 = pVIn->at(nSize - 1).nGetY();
505 nX1 = pVIn->at(0).nGetX();
506 nY1 = pVIn->at(0).nGetY();
507
508 dA = (nX0 * nY1) - (nX1 * nY0);
509 dSignedArea += dA;
510 PtiCentroid.AddXAddY((nX0 + nX1) * dA, (nY0 + nY1) * dA);
511
512 dSignedArea *= 0.5;
513 PtiCentroid.DivXDivY(6.0 * dSignedArea, 6.0 * dSignedArea);
514
515 return PtiCentroid;
516}

656double CSimulation::dAngleSubtended(CGeom2DIPoint const* pPtiA, CGeom2DIPoint const* pPtiB, CGeom2DIPoint const* pPtiC)
657{

659 dXDistBtoA = pPtiB->nGetX() - pPtiA->nGetX(),
660 dYDistBtoA = pPtiB->nGetY() - pPtiA->nGetY(),

662 dYDistCtoA = pPtiC->nGetY() - pPtiA->nGetY(),
663 dDotProduct = dXDistBtoA * dXDistCtoA + dYDistBtoA * dYDistCtoA,
664 dPseudoCrossProduct = dXDistBtoA * dYDistCtoA - dYDistBtoA * dXDistCtoA,
665 dAngle = atan2(dPseudoCrossProduct, dDotProduct);
666
667 return dAngle;
668}

703 return false;
704 }
705
706 // This driver is OK, so store its longname and the default file extension
707 string strTmp = CSLFetchNameValue(papszMetadata, "DMD_LONGNAME");
708 m_strGDALRasterOutputDriverLongname = strTrim(&strTmp);
709 strTmp = CSLFetchNameValue(papszMetadata, "DMD_EXTENSIONS"); // Note DMD_EXTENSION (no S, is a single value) appears not to be implemented for newer drivers

712 // We have a space-separated list of one or more file extensions: use the first extension in the list
713 long unsigned int nPos = strTmp.find(SPACE);
714 if (nPos == string::npos)
715 {
716 // No space i.e. just one extension
717 m_strGDALRasterOutputDriverExtension = strTmp;
718 }

721 // There's a space, so we must have more than one extension

753 m_GDALWriteIntDataType = GDT_UInt32;
754 m_lGDALMaxCanWrite = UINT32_MAX;
755 m_lGDALMinCanWrite = 0;
756

758 m_GDALWriteFloatDataType = GDT_UInt32;

760 return true;
761 }
762
763 if (strstr(CSLFetchNameValue(papszMetadata, "DMD_CREATIONDATATYPES"), "Int32"))
764 {
765 m_bGDALCanWriteInt32 = true;
766
767 m_GDALWriteIntDataType = GDT_Int32;
768 m_lGDALMaxCanWrite = INT32_MAX;
769 m_lGDALMinCanWrite = INT32_MIN;
770
771 if (! m_bGDALCanWriteFloat)
772 m_GDALWriteFloatDataType = GDT_Int32;
773
774 return true;
775 }
776
777 if (strstr(CSLFetchNameValue(papszMetadata, "DMD_CREATIONDATATYPES"), "UInt16"))
778 {
779 m_bGDALCanWriteInt32 = false;
780
781 m_GDALWriteIntDataType = GDT_UInt16;
782 m_lGDALMaxCanWrite = UINT16_MAX;
783 m_lGDALMinCanWrite = 0;
784
785 if (! m_bGDALCanWriteFloat)
786 m_GDALWriteFloatDataType = GDT_UInt16;
787
788 return true;
789 }
790
791 if (strstr(CSLFetchNameValue(papszMetadata, "DMD_CREATIONDATATYPES"), "Int16"))
792 {
793 m_bGDALCanWriteInt32 = false;
794
795 m_GDALWriteIntDataType = GDT_Int16;
796 m_lGDALMaxCanWrite = INT16_MAX;
797 m_lGDALMinCanWrite = INT16_MIN;
798
799 if (! m_bGDALCanWriteFloat)
800 m_GDALWriteFloatDataType = GDT_Int16;

808
809 m_GDALWriteIntDataType = GDT_Byte;

811 m_lGDALMinCanWrite = 0;

821 return false;
