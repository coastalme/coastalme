45double CSimulation::dCalculateSlope(int const nX1, int const nY1, int const nX2, int const nY2) const
46{
47 // Get top surface elevations
48 double const dElev1 = m_pRasterGrid->Cell(nX1, nY1).dGetSedimentTopElev();
49 double const dElev2 = m_pRasterGrid->Cell(nX2, nY2).dGetSedimentTopElev();
50
51 // Calculate distance between cells
52 double dDistance;
53 if ((nX1 != nX2) && (nY1 != nY2))
54 {
55 // Diagonal neighbor: distance is sqrt(2) * cell size
56 dDistance = 1.414213562 * m_dCellSide; // sqrt(2) â‰ˆ 1.414213562
57 }
58 else
59 {
60 // Orthogonal neighbor: distance is cell size
61 dDistance = m_dCellSide;
62 }
63
64 // Calculate slope as rise over run (tangent of angle)
65 double const dRise = dElev1 - dElev2;
66 double const dSlope = dRise / dDistance;
67
68 return dSlope;
69}

74double CSimulation::dCalculateInstability(int const nX, int const nY) const
75{
76 double dMaxInstability = 0.0;
77
78 // Check slopes to all 8 neighbors
79 for (int di = -1; di <= 1; di++)
80 {
81 for (int dj = -1; dj <= 1; dj++)
82 {
83 if (di == 0 && dj == 0)
84 continue; // Skip self
85
86 int const nNeighborX = nX + di;
87 int const nNeighborY = nY + dj;
88
89 // Check if neighbor is within grid bounds
90 if (nNeighborX < 0 || nNeighborX >= m_nXGridSize ||
91 nNeighborY < 0 || nNeighborY >= m_nYGridSize)
92 continue;
93
94 // Calculate slope to this neighbor
95 double const dSlope = dCalculateSlope(nX, nY, nNeighborX, nNeighborY);
96
97 // Calculate instability (excess slope beyond angle of repose)
98 double const dInstability = dSlope - TAN_ANGLE_OF_REPOSE;
99
100 // Track maximum instability
101 if (dInstability > dMaxInstability)
102 dMaxInstability = dInstability;
103 }
104 }
105
106 return dMaxInstability;
107}

112std::set<std::pair<int, int>> CSimulation::RedistributeSediment(int const nX, int const nY)
113{
114 std::set<std::pair<int, int>> affected_neighbors;
115
116 // Get source cell
117 CGeomCell& source_cell = m_pRasterGrid->Cell(nX, nY);
118
119 // Check if cell has sediment layers
120 int const nTopLayer = source_cell.nGetTopLayerAboveBasement();
121 if (nTopLayer < 0)
122 return affected_neighbors; // No layers to redistribute
123
124 // Get source cell elevation
125 double const dSourceElev = source_cell.dGetSedimentTopElev();
126
127 // Get unconsolidated sediment depths from top layer
128 CRWCellSediment* pUnconsolidated = source_cell.pGetLayerAboveBasement(nTopLayer)->pGetUnconsolidatedSediment();
129 double dSand = pUnconsolidated->dGetSandDepth();
130 double dCoarse = pUnconsolidated->dGetCoarseDepth();
131 double dFine = pUnconsolidated->dGetFineDepth();
132
133 double const dTotalDepth = dSand + dCoarse + dFine;
134
135 // Check if there's enough sediment to redistribute
136 if (dTotalDepth < MIN_AVALANCHE_VOLUME)
137 return affected_neighbors;
138
139 // Find downslope neighbors and calculate weights
140 struct Neighbor
141 {
142 int x;
143 int y;
144 double slope;
145 double weight;
146 };
147
148 std::vector<Neighbor> downslope_neighbors;
149 double dTotalWeight = 0.0;
150
151 for (int di = -1; di <= 1; di++)
152 {
153 for (int dj = -1; dj <= 1; dj++)
154 {
155 if (di == 0 && dj == 0)
156 continue;
157
158 int const nNeighborX = nX + di;
159 int const nNeighborY = nY + dj;
160
161 // Check bounds
162 if (nNeighborX < 0 || nNeighborX >= m_nXGridSize ||
163 nNeighborY < 0 || nNeighborY >= m_nYGridSize)
164 continue;
165
166 // Get neighbor elevation
167 double const dNeighborElev = m_pRasterGrid->Cell(nNeighborX, nNeighborY).dGetSedimentTopElev();
168
169 // Only redistribute to lower neighbors with excessive slope
170 if (dNeighborElev < dSourceElev)
171 {
172 double const dSlope = dCalculateSlope(nX, nY, nNeighborX, nNeighborY);
173
174 if (dSlope > TAN_ANGLE_OF_REPOSE)
175 {
176 // Weight proportional to excess slope
177 double const dWeight = dSlope - TAN_ANGLE_OF_REPOSE;
178 downslope_neighbors.push_back({nNeighborX, nNeighborY, dSlope, dWeight});
179 dTotalWeight += dWeight;
180 }
181 }
182 }
183 }
184
185 // No valid downslope neighbors
186 if (downslope_neighbors.empty())
187 return affected_neighbors;
188
189 // Calculate amount to redistribute
190 double const dAmountToMove = dTotalDepth * REDISTRIBUTION_FRACTION;
191
192 // Track remaining sediment in source cell
193 double dRemainingSand = dSand;
194 double dRemainingCoarse = dCoarse;
195 double dRemainingFine = dFine;
196
197 // Distribute proportionally to neighbors
198 for (auto const& neighbor : downslope_neighbors)
199 {
200 double const dFraction = neighbor.weight / dTotalWeight;
201 double const dMoveDepth = dAmountToMove * dFraction;
202
203 // Calculate amount of each size class to move (proportional to composition)
204 double const dMoveSand = dMoveDepth * (dSand / dTotalDepth);
205 double const dMoveCoarse = dMoveDepth * (dCoarse / dTotalDepth);
206 double const dMoveFine = dMoveDepth * (dFine / dTotalDepth);
207
208 // Update remaining amounts
209 dRemainingSand -= dMoveSand;
210 dRemainingCoarse -= dMoveCoarse;
211 dRemainingFine -= dMoveFine;
212
213 // Track avalanche deposition (total depth moved into this cell)
214 CGeomCell& neighbor_cell = m_pRasterGrid->Cell(neighbor.x, neighbor.y);
215 int const nNeighborTopLayer = neighbor_cell.nGetTopLayerAboveBasement();
216
217 if (nNeighborTopLayer >= 0)
218 {
219 CRWCellSediment* pNeighborUnconsolidated =
220 neighbor_cell.pGetLayerAboveBasement(nNeighborTopLayer)->pGetUnconsolidatedSediment();
221
222 pNeighborUnconsolidated->AddSandDepth(dMoveSand);
223 pNeighborUnconsolidated->AddCoarseDepth(dMoveCoarse);
224 // AddFineDepth() doesn't exist, so use SetFineDepth with current + new
225 double const dCurrentFine = pNeighborUnconsolidated->dGetFineDepth();
226 pNeighborUnconsolidated->SetFineDepth(dCurrentFine + dMoveFine);
227
228 // Recalculate neighbor elevations
229 neighbor_cell.CalcAllLayerElevsAndD50();
230 neighbor_cell.SetSeaDepth();
231
232 // Track avalanche deposition (total depth moved into this cell)
233 neighbor_cell.IncrAvalancheDeposition(dMoveDepth);
234
235 // Debug output
236 if (dMoveDepth > 0.001)
237 {
238 LogStream << " Avalanche: moved " << dMoveDepth << " m from ("
239 << nX << "," << nY << ") to (" << neighbor.x << "," << neighbor.y
240 << "), cell now has " << neighbor_cell.dGetAvalancheDeposition() << " m" << endl;
241 }
242
243 // Track affected neighbor
244 affected_neighbors.insert(std::make_pair(neighbor.x, neighbor.y));
245 MarkCellDirty(neighbor.x, neighbor.y);
246 }
247 }
248
249 // Update source cell with remaining sediment
250 pUnconsolidated->SetSandDepth(dRemainingSand);
251 pUnconsolidated->SetCoarseDepth(dRemainingCoarse);
252 pUnconsolidated->SetFineDepth(dRemainingFine);
253
254 // Recalculate source cell elevations
255 source_cell.CalcAllLayerElevsAndD50();
256 source_cell.SetSeaDepth();
257
258 return affected_neighbors;
259}

264int CSimulation::nDoSedimentAvalanching(void)
265{
266 // If no cells changed, nothing to do
267 if (m_DirtyCells.empty())
268 return RTN_OK;
269
270 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
271 {
272 LogStream << m_ulIter << ": Processing sediment avalanches on "
273 << m_DirtyCells.size() << " changed cells" << endl;
274 }
275
276 // Build initial candidate set: dirty cells + their 8 neighbors
277 std::set<std::pair<int, int>> candidates;
278
279 for (auto const& dirty_cell : m_DirtyCells)
280 {
281 candidates.insert(dirty_cell);
282
283 // Add 8 neighbors
284 for (int di = -1; di <= 1; di++)
285 {
286 for (int dj = -1; dj <= 1; dj++)
287 {
288 int const nNeighborX = dirty_cell.first + di;
289 int const nNeighborY = dirty_cell.second + dj;
290
291 if (nNeighborX >= 0 && nNeighborX < m_nXGridSize &&
292 nNeighborY >= 0 && nNeighborY < m_nYGridSize)
293 {
294 candidates.insert(std::make_pair(nNeighborX, nNeighborY));
295 }
296 }
297 }
298 }
299
300 // Build priority queue of unstable cells
301 std::priority_queue<UnstableCell, std::vector<UnstableCell>, UnstableCellComparator> pq;
302
303 for (auto const& cell : candidates)
304 {
305 double const dInstability = dCalculateInstability(cell.first, cell.second);
306 if (dInstability > 0.0)
307 {
308 pq.push(UnstableCell(cell.first, cell.second, dInstability));
309 }
310 }
311
312 // Process queue until empty or max iterations reached
313 int nIterations = 0;
314 int nCellsProcessed = 0;
315 std::set<std::pair<int, int>> processed_cells;
316
317 while (!pq.empty() && nIterations < MAX_AVALANCHE_ITERATIONS)
318 {
319 UnstableCell const unstable = pq.top();
320 pq.pop();
321
322 // Skip if already processed (may be in queue multiple times)
323 std::pair<int, int> const cell_coords = std::make_pair(unstable.nX, unstable.nY);
324 if (processed_cells.count(cell_coords) > 0)
325 continue;
326
327 // Redistribute sediment
328 std::set<std::pair<int, int>> const affected_neighbors =
329 RedistributeSediment(unstable.nX, unstable.nY);
330
331 // Mark as processed
332 processed_cells.insert(cell_coords);
333 nCellsProcessed++;
334
335 // Check affected neighbors for new instability
336 for (auto const& neighbor : affected_neighbors)
337 {
338 if (processed_cells.count(neighbor) == 0)
339 {
340 double const dNewInstability = dCalculateInstability(neighbor.first, neighbor.second);
341 if (dNewInstability > 0.0)
342 {
343 pq.push(UnstableCell(neighbor.first, neighbor.second, dNewInstability));
344 }
345 }
346 }
347
348 nIterations++;
349 }
350
351 // Log results
352 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
353 {
354 LogStream << " Processed " << nCellsProcessed << " unstable cells in "
355 << nIterations << " iterations" << endl;
356 }
357
358 // Warn if we hit max iterations
359 if (nIterations >= MAX_AVALANCHE_ITERATIONS)
360 {
361 LogStream << WARN << m_ulIter << ": Sediment avalanching hit maximum iteration limit ("
362 << MAX_AVALANCHE_ITERATIONS << ")" << endl;
363 }
364
365 // Note: m_DirtyCells is NOT cleared here, so it remains available for GIS output
366 // It will be cleared at the start of the next timestep
367
368 return RTN_OK;
369}
