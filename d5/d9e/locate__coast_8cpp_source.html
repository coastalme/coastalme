46int CSimulation::nLocateSeaAndCoasts(void)
47{
48 // Find all connected sea cells
49 FindAllSeaCells();
50
51 // Find every coastline on the raster grid, mark raster cells, then create the vector coastline
52 int const nRet = nTraceAllCoasts();
53 if (nRet != RTN_OK)
54 return nRet;
55
56 // Have we created any coasts?
57 if (m_VCoast.empty())
58 {
59 cerr << m_ulIter << ": " << ERR << "no coastline located: this iteration SWL = " << m_dThisIterSWL << ", maximum DEM top surface elevation = " << m_dThisIterTopElevMax << ", minimum DEM top surface elevation = " << m_dThisIterTopElevMin << endl;
60
61 return RTN_ERR_NO_COAST;
62 }
63
64 // Is this the highest SWL so far? If so, save the coast(s)
65 if (m_bHighestSWLSoFar)
66 {
67 m_VHighestSWLCoastLine.clear();
68 CGeomLine LCoast;
69 for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
70 {
71 LCoast = *m_VCoast[nCoast].pLGetCoastlineExtCRS();
72 m_VHighestSWLCoastLine.push_back(LCoast);
73 }
74 }
75
76 // Is this the lowest SWL so far? If so, save the coast(s)
77 if (m_bLowestSWLSoFar)
78 {
79 m_VLowestSWLCoastLine.clear();
80 CGeomLine LCoast;
81 for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
82 {
83 LCoast = *m_VCoast[nCoast].pLGetCoastlineExtCRS();
84 m_VLowestSWLCoastLine.push_back(LCoast);
85 }
86 }
87
88 return RTN_OK;
89}

94void CSimulation::FindAllSeaCells(void)
95{
96 // Go along the list of edge cells
97 for (unsigned int n = 0; n < m_VEdgeCell.size(); n++)
98 {
99 if (m_bOmitSearchNorthEdge && m_VEdgeCellEdge[n] == NORTH)
100 continue;
101
102 if (m_bOmitSearchSouthEdge && m_VEdgeCellEdge[n] == SOUTH)
103 continue;
104
105 if (m_bOmitSearchWestEdge && m_VEdgeCellEdge[n] == WEST)
106 continue;
107
108 if (m_bOmitSearchEastEdge && m_VEdgeCellEdge[n] == EAST)
109 continue;
110
111 int const nX = m_VEdgeCell[n].nGetX();
112 int const nY = m_VEdgeCell[n].nGetY();
113
114 if ((m_pRasterGrid->m_Cell[nX][nY].bIsInundated()) && (bFPIsEqual(m_pRasterGrid->m_Cell[nX][nY].dGetSeaDepth(), 0.0, TOLERANCE)))
115 // This edge cell is below SWL and sea depth remains set to zero
116 CellByCellFillSea(nX, nY);
117 }

306int CSimulation::nTraceAllCoasts(void)
307{
308 if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
309 LogStream << m_ulIter << ": Tracing coast" << endl;
310
311 int nValidCoast = 0;
312 vector<bool> VbPossibleStartCellLHEdge;
313 vector<bool> VbTraced;
314 vector<int> VnSearchDirection;
315 vector<CGeom2DIPoint> V2DIPossibleStartCell;
316
317 // Go along the list of edge cells and look for possible coastline start/finish cells
318 for (unsigned int n = 0; n < m_VEdgeCell.size() - 1; n++)
319 {
320 if (m_bOmitSearchNorthEdge && (m_VEdgeCellEdge[n] == NORTH || m_VEdgeCellEdge[n + 1] == NORTH))
321 continue;
322
323 if (m_bOmitSearchSouthEdge && (m_VEdgeCellEdge[n] == SOUTH || m_VEdgeCellEdge[n + 1] == SOUTH))
324 continue;
325
326 if (m_bOmitSearchWestEdge && (m_VEdgeCellEdge[n] == WEST || m_VEdgeCellEdge[n + 1] == WEST))
327 continue;
328
329 if (m_bOmitSearchEastEdge && (m_VEdgeCellEdge[n] == EAST || m_VEdgeCellEdge[n + 1] == EAST))
330 continue;
331
332 int const nXThis = m_VEdgeCell[n].nGetX();
333 int const nYThis = m_VEdgeCell[n].nGetY();
334 int const nXNext = m_VEdgeCell[n + 1].nGetX();
335 int const nYNext = m_VEdgeCell[n + 1].nGetY();
336
337 // Get "Is it sea?" information for 'this' and 'next' cells
338 bool const bThisCellIsSea = m_pRasterGrid->m_Cell[nXThis][nYThis].bIsInContiguousSea();
339 bool const bNextCellIsSea = m_pRasterGrid->m_Cell[nXNext][nYNext].bIsInContiguousSea();
340
341 // Are we at a coast?
342 if ((! bThisCellIsSea) && bNextCellIsSea)
343 {
344 // 'This' cell is just inland, has it already been flagged as a possible start for a coastline (even if this subsequently 'failed' as a coastline)?
345 if (! m_pRasterGrid->m_Cell[nXThis][nYThis].bIsPossibleCoastStartCell())
346 {
347 // It has not, so flag it
348 m_pRasterGrid->m_Cell[nXThis][nYThis].SetPossibleCoastStartCell();
349
350 if (m_nLogFileDetail >= LOG_FILE_ALL)
351 LogStream << m_ulIter << ": flagging [" << nXThis << "][" << nYThis << "] = {" << dGridCentroidXToExtCRSX(nXThis) << ", " << dGridCentroidYToExtCRSY(nYThis) << "} as possible coast start cell (left_handed edge)" << endl;
352
353 // And save it
354 V2DIPossibleStartCell.push_back(CGeom2DIPoint(nXThis, nYThis));
355 VbPossibleStartCellLHEdge.push_back(true);
356 VnSearchDirection.push_back(nGetOppositeDirection(m_VEdgeCellEdge[n]));
357 VbTraced.push_back(false);
358 }
359 }
360 else if (bThisCellIsSea && (! bNextCellIsSea))
361 {
362 // The 'next' cell is just inland, has it already been flagged as a possible start for a coastline (even if this subsequently 'failed' as a coastline)?
363 if (! m_pRasterGrid->m_Cell[nXNext][nYNext].bIsPossibleCoastStartCell())
364 {
365 // It has not, so flag it
366 m_pRasterGrid->m_Cell[nXNext][nYNext].SetPossibleCoastStartCell();
367
368 if (m_nLogFileDetail >= LOG_FILE_ALL)
369 LogStream << m_ulIter << ": flagging [" << nXNext << "][" << nYNext << "] = {" << dGridCentroidXToExtCRSX(nXNext) << ", " << dGridCentroidYToExtCRSY(nYNext) << "} as possible coast start cell (right_handed edge)" << endl;
370
371 // And save it
372 V2DIPossibleStartCell.push_back(CGeom2DIPoint(nXNext, nYNext));
373 VbPossibleStartCellLHEdge.push_back(false);
374 VnSearchDirection.push_back(nGetOppositeDirection(m_VEdgeCellEdge[n + 1]));
375 VbTraced.push_back(false);
376 }
377 }
378 }
379
380 // Any possible coastline start/finish cells found?
381 if (V2DIPossibleStartCell.size() == 0)
382 {
383 LogStream << m_ulIter << ": no coastline start/finish points found after grid edges searched.";
384
385 if (m_bOmitSearchNorthEdge || m_bOmitSearchSouthEdge || m_bOmitSearchWestEdge || m_bOmitSearchEastEdge)
386 {
387 LogStream << " Note that the following grid edges were not searched: " << (m_bOmitSearchNorthEdge ? "N " : "") << (m_bOmitSearchSouthEdge ? "S " : "") << (m_bOmitSearchWestEdge ? "W " : "") << (m_bOmitSearchEastEdge ? "E " : "");
388 }
389
390 LogStream << endl;
391
392 return RTN_ERR_NO_START_FINISH_POINTS_TRACING_COAST;
393 }
394
395 // Some possible coastline start/finish points were found
396 // LogStream << m_ulIter << ": " << V2DIPossibleStartCell.size() << " possible coastline start/finish points found" << endl;
397
398 // OK, so trace from each of these possible start/finish points
399 for (unsigned int n = 0; n < V2DIPossibleStartCell.size(); n++)
400 {
401 if (! VbTraced[n])
402 {
403 int nRet = 0;
404
405 if (VbPossibleStartCellLHEdge[n])
406 nRet = nTraceCoastLine(n, VnSearchDirection[n], LEFT_HANDED, &VbTraced, &V2DIPossibleStartCell);
407 else
408 nRet = nTraceCoastLine(n, VnSearchDirection[n], RIGHT_HANDED, &VbTraced, &V2DIPossibleStartCell);
409
410 if (nRet == RTN_OK)
411 {
412 // We have a valid coastline starting from this possible start cell
413 VbTraced[n] = true;
414 nValidCoast++;
415 }
416 }
417 }
418
419 if (nValidCoast > 0)
420 return RTN_OK;
421
422 cerr << m_ulIter << ": no valid coasts found, see " << m_strLogFile << " for more information" << endl;
423 return RTN_ERR_NO_VALID_COAST;
424}
