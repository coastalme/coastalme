1482template <class T> bool bFPIsEqual(const T d1, const T d2, const T dEpsilon) {
1483 // Since the accuracy of floating-point numbers varies with their magnitude,
1484 // we must compare them by using an accuracy threshold which is relative to
1485 // the magnitude of the two numbers being compared. This is a blend of an
1486 // example from Knuth's 'The Art of Computer Programming. Volume 1.
1487 // Fundamental Algorithms' and a posting dated 18 Nov 93 by rmartin@rcmcon.com
1488 // (Robert Martin), archived in cpp_tips
1489
1490#pragma GCC diagnostic push
1491#pragma GCC diagnostic ignored "-Wfloat-equal"
1492 if ((0 == d1) && (tAbs(d2) < dEpsilon))
1493 return true;
1494
1495 else if ((0 == d2) && (tAbs(d1) < dEpsilon))
1496 return true;
1497
1498 else
1499 return ((tAbs(d1 - d2) < (dEpsilon * tAbs(d1))) ? true : false);
1500#pragma GCC diagnostic pop
1501}
