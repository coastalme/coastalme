1191// For comparison of two floating-point numbers, with a specified accuracy
1192// ==============================================================================================================================
1193template <class T>
1194bool bFPIsEqual(const T d1, const T d2, const T dEpsilon)
1195{
1196 // Since the accuracy of floating-point numbers varies with their magnitude, we must compare them by using an accuracy threshold which is relative to the magnitude of the two numbers being compared. This is a blend of an example from Knuth's 'The Art of Computer Programming. Volume 1. Fundamental Algorithms' and a posting dated 18 Nov 93 by rmartin@rcmcon.com (Robert Martin), archived in cpp_tips
1197
1198#pragma GCC diagnostic push
1199#pragma GCC diagnostic ignored "-Wfloat-equal"
1200
1201 if ((0 == d1) && (tAbs(d2) < dEpsilon))
1202 return true;
1203
1204 else if ((0 == d2) && (tAbs(d1) < dEpsilon))
1205 return true;
1206
1207 else
1208 return ((tAbs(d1 - d2) < (dEpsilon * tAbs(d1))) ? true : false);
1209
1210#pragma GCC diagnostic pop
1211}
