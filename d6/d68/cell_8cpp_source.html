709void CGeomCell::CalcAllLayerElevsAndD50(void)
710{
711 m_VdAllHorizonTopElev.clear();
712 m_VdAllHorizonTopElev.push_back(m_dBasementElevation); // Elevation of top of the basement
713
714 // Calculate the elevation of the top of all other layers
715 int m = 0;
716
717 for (unsigned int n = 0; n < m_VLayerAboveBasement.size(); n++)
718 m_VdAllHorizonTopElev.push_back(m_VLayerAboveBasement[n].dGetTotalThickness() + m_VdAllHorizonTopElev[m++]); // Elevation of top of layer n
719
720 // Now calculate the d50 of the topmost unconsolidated sediment layer with non-zero thickness. If there is no unconsolidated sediment, m_dUnconsD50 is set to DBL_NODATA
721 m_dUnconsD50 = DBL_NODATA;
722
723 for (int n = static_cast<int>(m_VLayerAboveBasement.size()) - 1; n >= 0; n--)
724 {
725 double dUnconsThick = m_VLayerAboveBasement[n].dGetUnconsolidatedThickness();
726
727 if (dUnconsThick > 0)
728 {
729 // This is a layer with non-zero thickness of unconsolidated sediment
730 CRWCellSediment const* pUnconsSedLayer = m_VLayerAboveBasement[n].pGetUnconsolidatedSediment();
731 double dFineProp = pUnconsSedLayer->dGetFineDepth() / dUnconsThick;
732 double dSandProp = pUnconsSedLayer->dGetSandDepth() / dUnconsThick;
733 double dCoarseProp = pUnconsSedLayer->dGetCoarseDepth() / dUnconsThick;
734
735 // Calculate d50 for the unconsolidated sediment
736 m_dUnconsD50 = (dFineProp * m_pGrid->pGetSim()->dGetD50Fine()) + (dSandProp * m_pGrid->pGetSim()->dGetD50Sand()) + (dCoarseProp * m_pGrid->pGetSim()->dGetD50Coarse());
737
738 break;
739 }
740 }
741}

744int CGeomCell::nGetLayerAtElev(double const dElev) const
745{
747 if (dElev < m_VdAllHorizonTopElev[0])
748 return ELEV_IN_BASEMENT;
749
750 for (unsigned int nLayer = 1; nLayer < m_VdAllHorizonTopElev.size(); nLayer++)
751 {
752 if ((m_VLayerAboveBasement[nLayer - 1].dGetTotalThickness() > 0) && (dElev >= m_VdAllHorizonTopElev[nLayer - 1]) && (dElev <= m_VdAllHorizonTopElev[nLayer]))
753 return (nLayer - 1);
754 }
755
756 return ELEV_ABOVE_SEDIMENT_TOP;
757}
