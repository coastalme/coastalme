791void CGeomCell::CalcAllLayerElevsAndD50(void)
792{
793 m_VdAllHorizonTopElev.clear();
794 m_VdAllHorizonTopElev.push_back(m_dBasementElevation); // Elevation of top of the basement
795
796 // Calculate the elevation of the top of all other layers
797 int m = 0;
798
799 for (unsigned int n = 0; n < m_VLayerAboveBasement.size(); n++)
800 m_VdAllHorizonTopElev.push_back(m_VLayerAboveBasement[n].dGetTotalThickness() + m_VdAllHorizonTopElev[m++]); // Elevation of top of layer n
801
802 // Now calculate the d50 of the topmost unconsolidated sediment layer with non-zero thickness. If there is no unconsolidated sediment, m_dUnconsD50 is set to DBL_NODATA
803 m_dUnconsD50 = DBL_NODATA;
804
805 for (int n = static_cast<int>(m_VLayerAboveBasement.size()) - 1; n >= 0; n--)
806 {
807 double const dUnconsThick = m_VLayerAboveBasement[n].dGetUnconsolidatedThickness();
808
809 if (dUnconsThick > 0)
810 {
811 // This is a layer with non-zero thickness of unconsolidated sediment
812 CRWCellSediment const *pUnconsSedLayer = m_VLayerAboveBasement[n].pGetUnconsolidatedSediment();
813 double const dFineProp = pUnconsSedLayer->dGetFineDepth() / dUnconsThick;
814 double const dSandProp = pUnconsSedLayer->dGetSandDepth() / dUnconsThick;
815 double const dCoarseProp = pUnconsSedLayer->dGetCoarseDepth() / dUnconsThick;
816
