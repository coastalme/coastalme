695void CGeomCell::CalcAllLayerElevsAndD50(void)
696{
697 m_VdAllHorizonTopElev.clear();
698 m_VdAllHorizonTopElev.push_back(m_dBasementElevation); // Elevation of top of the basement
699
700 // Calculate the elevation of the top of all other layers
701 int m = 0;
702 for (unsigned int n = 0; n < m_VLayerAboveBasement.size(); n++)
703 m_VdAllHorizonTopElev.push_back(m_VLayerAboveBasement[n].dGetTotalThickness() + m_VdAllHorizonTopElev[m++]); // Elevation of top of layer n
704
705 // Now calculate the d50 of the topmost unconsolidated sediment layer with non-zero thickness. If there is no unconsolidated sediment, m_dUnconsD50 is set to DBL_NODATA
706 m_dUnconsD50 = DBL_NODATA;
707 for (int n = static_cast<int>(m_VLayerAboveBasement.size()) - 1; n >= 0; n--)
708 {
709 double dUnconsThick = m_VLayerAboveBasement[n].dGetUnconsolidatedThickness();
710 if (dUnconsThick > 0)
711 {
712 // This is a layer with non-zero thickness of unconsolidated sediment
713 CRWCellSediment const* pUnconsSedLayer = m_VLayerAboveBasement[n].pGetUnconsolidatedSediment();
714 double dFineProp = pUnconsSedLayer->dGetFineDepth() / dUnconsThick;
715 double dSandProp = pUnconsSedLayer->dGetSandDepth() / dUnconsThick;
716 double dCoarseProp = pUnconsSedLayer->dGetCoarseDepth() / dUnconsThick;
717
718 // Calculate d50 for the unconsolidated sediment
719 m_dUnconsD50 = (dFineProp * m_pGrid->pGetSim()->dGetD50Fine()) + (dSandProp * m_pGrid->pGetSim()->dGetD50Sand()) + (dCoarseProp * m_pGrid->pGetSim()->dGetD50Coarse());
720
721 break;
722 }
723 }
724}

727int CGeomCell::nGetLayerAtElev(double const dElev) const
728{
730 if (dElev < m_VdAllHorizonTopElev[0])
731 return ELEV_IN_BASEMENT;
732
733 for (unsigned int nLayer = 1; nLayer < m_VdAllHorizonTopElev.size(); nLayer++)
734 {
735 if ((m_VLayerAboveBasement[nLayer - 1].dGetTotalThickness() > 0) && (dElev >= m_VdAllHorizonTopElev[nLayer - 1]) && (dElev <= m_VdAllHorizonTopElev[nLayer]))
736 return (nLayer - 1);
737 }
738
739 return ELEV_ABOVE_SEDIMENT_TOP;
740}
