258int CRWCoast::nGetCoastPointGivenCell(CGeom2DIPoint* pPtiCell)
259{
260 for (int nCoastPoint = 0; nCoastPoint < m_ILCellsMarkedAsCoastline.nGetSize(); nCoastPoint++)
261 {
262 if (m_ILCellsMarkedAsCoastline[nCoastPoint] == pPtiCell)
263 {
264 return nCoastPoint;
265 }
266 }
267
268 // This cell is not under a coastline, so try the adjacent cells
269 int
270 n = -1,
271 nX = pPtiCell->nGetX(),
272 nY = pPtiCell->nGetY(),
273 nXAdj = 0,
274 nYAdj = 0;
275
276 while (n <= 7)
277 {
278 switch (++n)
279 {
280 case 0:
281 nXAdj = nX;
282 nYAdj = nY - 1;
283 break;
284 case 1:
285 nXAdj = nX + 1;
286 nYAdj = nY - 1;
287 break;
288 case 2:
289 nXAdj = nX + 1;
290 nYAdj = nY;
291 break;
292 case 3:
293 nXAdj = nX + 1;
294 nYAdj = nY + 1;
295 break;
296 case 4:
297 nXAdj = nX;
298 nYAdj = nY + 1;
299 break;
300 case 5:
301 nXAdj = nX - 1;
302 nYAdj = nY + 1;
303 break;
304 case 6:
305 nXAdj = nX - 1;
306 nYAdj = nY;
307 break;
308 case 7:
309 nXAdj = nX - 1;
310 nYAdj = nY - 1;
311 break;
312 }
313
314 CGeom2DIPoint PtiTmp(nXAdj, nYAdj);
315 for (int nCoastPoint = 0; nCoastPoint < m_ILCellsMarkedAsCoastline.nGetSize(); nCoastPoint++)
316 {
317 if (m_ILCellsMarkedAsCoastline[nCoastPoint] == &PtiTmp)
318 {
319 *pPtiCell = PtiTmp;
320 return nCoastPoint;
321 }
322 }
323 }
324
325 return INT_NODATA;
326}

588double CRWCoast::dGetLevel(int const nCoastPoint, int const level) const
589{
590 switch (level)
591 {
592 case 0: // WAVESETUPSURGE:
593 return m_VdWaveSetupSurge[nCoastPoint];
