243int CRWCoast::nGetCoastPointGivenCell(CGeom2DIPoint* pPtiCell)
244{
245 for (int nCoastPoint = 0; nCoastPoint < m_ILCellsMarkedAsCoastline.nGetSize(); nCoastPoint++)
246 {
247 if (m_ILCellsMarkedAsCoastline[nCoastPoint] == pPtiCell)
248 {
249 return nCoastPoint;
250 }
251 }
252
253 // This cell is not under a coastline, so try the adjacent cells
254 int n = -1;
255 int nX = pPtiCell->nGetX();
256 int nY = pPtiCell->nGetY();
257 int nXAdj = 0;
258 int nYAdj = 0;
259
260 while (n <= 7)
261 {
262 switch (++n)
263 {
264 case 0:
265 nXAdj = nX;
266 nYAdj = nY - 1;
267 break;
268 case 1:
269 nXAdj = nX + 1;
270 nYAdj = nY - 1;
271 break;
272 case 2:
273 nXAdj = nX + 1;
274 nYAdj = nY;
275 break;
276 case 3:
277 nXAdj = nX + 1;
278 nYAdj = nY + 1;
279 break;
280 case 4:
281 nXAdj = nX;
282 nYAdj = nY + 1;
283 break;
284 case 5:
285 nXAdj = nX - 1;
286 nYAdj = nY + 1;
287 break;
288 case 6:
289 nXAdj = nX - 1;
290 nYAdj = nY;
291 break;
292 case 7:
293 nXAdj = nX - 1;
294 nYAdj = nY - 1;
295 break;
296 }
297
298 CGeom2DIPoint PtiTmp(nXAdj, nYAdj);
299 for (int nCoastPoint = 0; nCoastPoint < m_ILCellsMarkedAsCoastline.nGetSize(); nCoastPoint++)
300 {
301 if (m_ILCellsMarkedAsCoastline[nCoastPoint] == &PtiTmp)
302 {
303 *pPtiCell = PtiTmp;
304 return nCoastPoint;
305 }
306 }
307 }
308
309 return INT_NODATA;
310}
